# 创建TODO管理工具
from dataclasses import dataclass
from enum import Enum
from typing import List, Dict
from datetime import datetime

class TodoPriority(Enum):
    """TODO优先级"""
    CRITICAL = "critical"    # 阻塞功能
    HIGH = "high"           # 重要功能
    MEDIUM = "medium"       # 优化改进
    LOW = "low"            # 代码清理

@dataclass
class TodoItem:
    """TODO项目"""
    file_path: str
    line_number: int
    description: str
    priority: TodoPriority
    assignee: str = ""
    created_at: datetime = None
    estimated_hours: float = 0.0
    status: str = "pending"

class TodoManager:
    """TODO管理器"""

    def __init__(self):
        self.todos: List[TodoItem] = []
        self.patterns = {
            r"# TODO: (.+)": TodoPriority.MEDIUM,
            r"# FIXME: (.+)": TodoPriority.HIGH,
            r"# HACK: (.+)": TodoPriority.HIGH,
            r"# XXX: (.+)": TodoPriority.LOW,
            r"# BUG: (.+)": TodoPriority.CRITICAL,
        }

    def scan_project(self, project_path: str) -> List[TodoItem]:
        """扫描项目中的TODO"""
        import os
        import re

        todos = []

        for root, dirs, files in os.walk(project_path):
            for file in files:
                if file.endswith('.py'):
                    file_path = os.path.join(root, file)
                    todos.extend(self._scan_file(file_path))

        return todos

    def _scan_file(self, file_path: str) -> List[TodoItem]:
        """扫描单个文件"""
        todos = []

        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                lines = f.readlines()

            for line_num, line in enumerate(lines, 1):
                for pattern, priority in self.patterns.items():
                    match = re.search(pattern, line)
                    if match:
                        todo = TodoItem(
                            file_path=file_path,
                            line_number=line_num,
                            description=match.group(1).strip(),
                            priority=priority,
                            created_at=datetime.now()
                        )
                        todos.append(todo)
                        break

        except Exception as e:
            print(f"扫描文件失败 {file_path}: {e}")

        return todos

    def generate_report(self) -> str:
        """生成TODO报告"""
        report = []
        report.append("=" * 80)
        report.append("VirtualChemLab TODO/FIXME 报告")
        report.append("=" * 80)
        report.append(f"扫描时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append(f"总计: {len(self.todos)} 个TODO项目")
        report.append("")

        # 按优先级分组
        by_priority = {}
        for todo in self.todos:
            if todo.priority not in by_priority:
                by_priority[todo.priority] = []
            by_priority[todo.priority].append(todo)

        # 按优先级排序
        priority_order = [TodoPriority.CRITICAL, TodoPriority.HIGH,
                         TodoPriority.MEDIUM, TodoPriority.LOW]

        for priority in priority_order:
            if priority in by_priority:
                todos = by_priority[priority]
                report.append(f"## {priority.value.upper()} 优先级 ({len(todos)} 个)")
                report.append("-" * 40)

                for todo in todos:
                    report.append(f"### {todo.file_path}:{todo.line_number}")
                    report.append(f"描述: {todo.description}")
                    report.append(f"状态: {todo.status}")
                    if todo.assignee:
                        report.append(f"负责人: {todo.assignee}")
                    if todo.estimated_hours > 0:
                        report.append(f"预估工时: {todo.estimated_hours}小时")
                    report.append("")

        return "\n".join(report)

```python
from typing import TypeVar, Callable, Any, Optional
from functools import wraps
import logging
from enum import Enum

T = TypeVar('T')

class ErrorStrategy(Enum):
    """错误处理策略"""
    RETRY = "retry"
    FALLBACK = "fallback"
    LOG_AND_CONTINUE = "log_and_continue"
    LOG_AND_RAISE = "log_and_raise"
    SILENT_FAIL = "silent_fail"

class ExceptionHandler:
    """统一异常处理器"""

    def __init__(self):
        self.logger = logging.getLogger(__name__)
        self.retry_configs = {}
        self.fallback_functions = {}

    def handle_exception(self,
                        exception: Exception,
                        context: str = "",
                        strategy: ErrorStrategy = ErrorStrategy.LOG_AND_RAISE,
                        retry_count: int = 0,
                        fallback_func: Optional[Callable] = None) -> Any:
        """处理异常"""

        # 记录异常
        self.logger.error(f"异常处理 [{context}]: {type(exception).__name__}: {exception}")

        if strategy == ErrorStrategy.RETRY and retry_count > 0:
            return self._handle_retry(exception, context, retry_count)

        elif strategy == ErrorStrategy.FALLBACK and fallback_func:
            return self._handle_fallback(exception, context, fallback_func)

        elif strategy == ErrorStrategy.LOG_AND_CONTINUE:
            return None

        elif strategy == ErrorStrategy.SILENT_FAIL:
            return None

        else:  # LOG_AND_RAISE
            raise exception

    def _handle_retry(self, exception: Exception, context: str, retry_count: int) -> Any:
        """处理重试"""
        import time

        self.logger.info(f"重试操作 [{context}], 剩余次数: {retry_count}")
        time.sleep(1)  # 简单的退避策略

        # 这里需要调用原始函数，实际实现会更复杂
        return None

    def _handle_fallback(self, exception: Exception, context: str, fallback_func: Callable) -> Any:
        """处理回退"""
        self.logger.info(f"使用回退函数 [{context}]")
        try:
            return fallback_func()
        except Exception as e:
            self.logger.error(f"回退函数也失败: {e}")
            raise exception

# 装饰器
def safe_execute(strategy: ErrorStrategy = ErrorStrategy.LOG_AND_RAISE,
                retry_count: int = 0,
                fallback_func: Optional[Callable] = None,
                context: str = ""):
    """安全执行装饰器"""
    def decorator(func: Callable[..., T]) -> Callable[..., Optional[T]]:
        @wraps(func)
        def wrapper(*args, **kwargs) -> Optional[T]:
            handler = ExceptionHandler()

            try:
                return func(*args, **kwargs)
            except Exception as e:
                return handler.handle_exception(
                    e,
                    context or func.__name__,
                    strategy,
                    retry_count,
                    fallback_func
                )

        return wrapper
    return decorator

# 使用示例
@safe_execute(
    strategy=ErrorStrategy.RETRY,
    retry_count=3,
    context="文件操作"
)
def read_file_safely(file_path: str) -> str:
    """安全读取文件"""
    with open(file_path, 'r') as f:
        return f.read()

@safe_execute(
    strategy=ErrorStrategy.FALLBACK,
    fallback_func=lambda: "默认值",
    context="配置读取"
)
def get_config_value(key: str) -> str:
    """获取配置值"""
    # 可能失败的操作
    return config[key]
```

```python
from typing import Dict, Type, Callable
import os
import json
import sqlite3
import requests

class SpecificExceptionHandler:
    """具体异常类型处理器"""

    def __init__(self):
        self.handlers: Dict[Type[Exception], Callable] = {
            FileNotFoundError: self._handle_file_not_found,
            PermissionError: self._handle_permission_error,
            json.JSONDecodeError: self._handle_json_error,
            sqlite3.OperationalError: self._handle_database_error,
            requests.RequestException: self._handle_network_error,
            ValueError: self._handle_value_error,
            TypeError: self._handle_type_error,
        }

    def handle(self, exception: Exception, context: str = "") -> Any:
        """处理具体异常"""
        exception_type = type(exception)

        if exception_type in self.handlers:
            return self.handlers[exception_type](exception, context)
        else:
            # 默认处理
            return self._handle_generic(exception, context)

    def _handle_file_not_found(self, exception: FileNotFoundError, context: str) -> Any:
        """处理文件未找到"""
        logger.error(f"文件未找到 [{context}]: {exception.filename}")

        # 尝试创建默认文件
        if context == "配置文件":
            return self._create_default_config(exception.filename)
        elif context == "日志文件":
            return self._create_log_directory(exception.filename)

        return None

    def _handle_permission_error(self, exception: PermissionError, context: str) -> Any:
        """处理权限错误"""
        logger.error(f"权限不足 [{context}]: {exception}")

        # 尝试使用备用路径
        if context == "数据文件":
            return self._use_backup_path()

        return None

    def _handle_json_error(self, exception: json.JSONDecodeError, context: str) -> Any:
        """处理JSON解析错误"""
        logger.error(f"JSON解析失败 [{context}]: {exception}")

        # 尝试修复JSON
        if context == "配置文件":
            return self._repair_json_file(exception)

        return None

    def _handle_database_error(self, exception: sqlite3.OperationalError, context: str) -> Any:
        """处理数据库错误"""
        logger.error(f"数据库操作失败 [{context}]: {exception}")

        # 尝试重建数据库
        if "no such table" in str(exception):
            return self._recreate_database()

        return None

    def _handle_network_error(self, exception: requests.RequestException, context: str) -> Any:
        """处理网络错误"""
        logger.error(f"网络请求失败 [{context}]: {exception}")

        # 使用缓存数据
        if context == "API请求":
            return self._use_cached_data()

        return None

    def _handle_value_error(self, exception: ValueError, context: str) -> Any:
        """处理值错误"""
        logger.error(f"值错误 [{context}]: {exception}")

        # 尝试转换类型
        if context == "配置值":
            return self._convert_config_value(exception)

        return None

    def _handle_type_error(self, exception: TypeError, context: str) -> Any:
        """处理类型错误"""
        logger.error(f"类型错误 [{context}]: {exception}")

        # 尝试类型转换
        if context == "数据处理":
            return self._convert_data_type(exception)

        return None

    def _handle_generic(self, exception: Exception, context: str) -> Any:
        """处理通用异常"""
        logger.error(f"未处理的异常 [{context}]: {type(exception).__name__}: {exception}")
        return None
```

#### 3.1 循环依赖解决

```python
# 创建依赖注入容器来解决循环依赖
from typing import Protocol, TypeVar, Generic
from abc import ABC, abstractmethod

T = TypeVar('T')

class ServiceProtocol(Protocol):
    """服务协议"""
    def initialize(self) -> None: ...
    def cleanup(self) -> None: ...

class DependencyContainer:
    """依赖注入容器"""

    def __init__(self):
        self._services: Dict[str, Any] = {}
        self._factories: Dict[str, Callable] = {}
        self._singletons: Dict[str, Any] = {}

    def register_service(self, name: str, service_class: Type[T], singleton: bool = True) -> None:
        """注册服务"""
        self._factories[name] = lambda: service_class()
        if singleton:
            self._singletons[name] = None

    def get_service(self, name: str) -> Any:
        """获取服务"""
        if name in self._singletons and self._singletons[name] is not None:
            return self._singletons[name]

        if name in self._factories:
            service = self._factories[name]()

            if name in self._singletons:
                self._singletons[name] = service

            return service

        raise ValueError(f"服务未注册: {name}")

    def resolve_dependencies(self, service_class: Type[T]) -> T:
        """解析依赖"""
        # 使用反射解析构造函数参数
        import inspect

        sig = inspect.signature(service_class.__init__)
        kwargs = {}

        for param_name, param in sig.parameters.items():
            if param_name == 'self':
                continue

            # 尝试从容器获取依赖
            if param_name in self._services:
                kwargs[param_name] = self._services[param_name]
            else:
                # 使用默认值
                if param.default != inspect.Parameter.empty:
                    kwargs[param_name] = param.default

        return service_class(**kwargs)

# 使用示例
class DatabaseService:
    def __init__(self, config_service: 'ConfigService'):
        self.config = config_service

class ConfigService:
    def __init__(self):
        self.settings = {}

class Application:
    def __init__(self):
        self.container = DependencyContainer()
        self._setup_services()

    def _setup_services(self):
        """设置服务"""
        # 注册服务
        self.container.register_service('config', ConfigService)
        self.container.register_service('database', DatabaseService)

        # 解析依赖
        self.config_service = self.container.get_service('config')
        self.database_service = self.container.get_service('database')
```

```python
from dataclasses import dataclass
from typing import Any, Dict, Optional
from pathlib import Path
import json
import yaml

@dataclass
class ConfigValue:
    """配置值"""
    value: Any
    default: Any
    description: str
    required: bool = False
    validator: Optional[Callable] = None

class ConfigManager:
    """配置管理器"""

    def __init__(self, config_file: Path):
        self.config_file = config_file
        self.config: Dict[str, Any] = {}
        self.schema: Dict[str, ConfigValue] = {}
        self._load_config()

    def define_config(self, key: str, default: Any, description: str = "",
                     required: bool = False, validator: Optional[Callable] = None) -> None:
        """定义配置项"""
        self.schema[key] = ConfigValue(
            value=default,
            default=default,
            description=description,
            required=required,
            validator=validator
        )

    def get(self, key: str, default: Any = None) -> Any:
        """获取配置值"""
        if key in self.config:
            return self.config[key]

        if key in self.schema:
            return self.schema[key].value

        return default

    def set(self, key: str, value: Any) -> None:
        """设置配置值"""
        if key in self.schema:
            config_value = self.schema[key]

            # 验证值
            if config_value.validator and not config_value.validator(value):
                raise ValueError(f"配置值验证失败: {key}")

            config_value.value = value

        self.config[key] = value

    def _load_config(self) -> None:
        """加载配置"""
        if not self.config_file.exists():
            self._create_default_config()
            return

        try:
            if self.config_file.suffix == '.json':
                with open(self.config_file, 'r', encoding='utf-8') as f:
                    self.config = json.load(f)
            elif self.config_file.suffix in ['.yml', '.yaml']:
                with open(self.config_file, 'r', encoding='utf-8') as f:
                    self.config = yaml.safe_load(f)
            else:
                raise ValueError(f"不支持的配置文件格式: {self.config_file.suffix}")

        except Exception as e:
            print(f"加载配置文件失败: {e}")
            self._create_default_config()

    def _create_default_config(self) -> None:
        """创建默认配置"""
        self.config = {}
        for key, config_value in self.schema.items():
            self.config[key] = config_value.default

        self.save_config()

    def save_config(self) -> None:
        """保存配置"""
        try:
            self.config_file.parent.mkdir(parents=True, exist_ok=True)

            if self.config_file.suffix == '.json':
                with open(self.config_file, 'w', encoding='utf-8') as f:
                    json.dump(self.config, f, indent=2, ensure_ascii=False)
            elif self.config_file.suffix in ['.yml', '.yaml']:
                with open(self.config_file, 'w', encoding='utf-8') as f:
                    yaml.dump(self.config, f, default_flow_style=False, allow_unicode=True)

        except Exception as e:
            print(f"保存配置文件失败: {e}")

    def validate_config(self) -> List[str]:
        """验证配置"""
        errors = []

        for key, config_value in self.schema.items():
            if config_value.required and key not in self.config:
                errors.append(f"缺少必需的配置项: {key}")

            if key in self.config:
                value = self.config[key]
                if config_value.validator and not config_value.validator(value):
                    errors.append(f"配置值验证失败: {key}")

        return errors

    def get_config_documentation(self) -> str:
        """获取配置文档"""
        doc = []
        doc.append("# 配置文档")
        doc.append("")

        for key, config_value in self.schema.items():
            doc.append(f"## {key}")
            doc.append(f"**描述**: {config_value.description}")
            doc.append(f"**默认值**: {config_value.default}")
            doc.append(f"**必需**: {'是' if config_value.required else '否'}")
            doc.append("")

        return "\n".join(doc)

# 使用示例
```

```python
import ast
import re
from typing import List, Dict, Any
from dataclasses import dataclass
from pathlib import Path

@dataclass
class CodeIssue:
    """代码问题"""
    file_path: str
    line_number: int
    issue_type: str
    message: str
    severity: str
    suggestion: str = ""

class CodeQualityChecker:
    """代码质量检查器"""

    def __init__(self):
        self.issues: List[CodeIssue] = []
        self.rules = {
            "broad_except": self._check_broad_except,
            "empty_except": self._check_empty_except,
            "hardcoded_values": self._check_hardcoded_values,
            "long_functions": self._check_long_functions,
            "complex_functions": self._check_complex_functions,
            "duplicate_code": self._check_duplicate_code,
            "unused_imports": self._check_unused_imports,
            "magic_numbers": self._check_magic_numbers,
        }

    def check_file(self, file_path: Path) -> List[CodeIssue]:
        """检查单个文件"""
        issues = []

        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()

            # 解析AST
            tree = ast.parse(content)

            # 应用所有规则
            for rule_name, rule_func in self.rules.items():
                rule_issues = rule_func(tree, file_path, content)
                issues.extend(rule_issues)

        except Exception as e:
            issues.append(CodeIssue(
                file_path=str(file_path),
                line_number=0,
                issue_type="parse_error",
                message=f"文件解析失败: {e}",
                severity="error"
            ))

        return issues

    def _check_broad_except(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查过于宽泛的异常捕获"""
        issues = []

        class BroadExceptVisitor(ast.NodeVisitor):
            def visit_ExceptHandler(self, node):
                if node.type is None or (isinstance(node.type, ast.Name) and node.type.id == 'Exception'):
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="broad_except",
                        message="使用了过于宽泛的异常捕获",
                        severity="warning",
                        suggestion="使用具体的异常类型，如 FileNotFoundError, ValueError 等"
                    ))
                self.generic_visit(node)

        BroadExceptVisitor().visit(tree)
        return issues

    def _check_empty_except(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查空的异常处理"""
        issues = []

        class EmptyExceptVisitor(ast.NodeVisitor):
            def visit_ExceptHandler(self, node):
                if not node.body or (len(node.body) == 1 and isinstance(node.body[0], ast.Pass)):
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="empty_except",
                        message="空的异常处理块",
                        severity="warning",
                        suggestion="添加适当的错误处理逻辑或记录日志"
                    ))
                self.generic_visit(node)

        EmptyExceptVisitor().visit(tree)
        return issues

    def _check_hardcoded_values(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查硬编码值"""
        issues = []

        # 检查数字字面量
        class HardcodedVisitor(ast.NodeVisitor):
            def visit_Num(self, node):
                if isinstance(node.n, int) and node.n > 100:
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="hardcoded_value",
                        message=f"硬编码数值: {node.n}",
                        severity="info",
                        suggestion="考虑将数值移到配置文件或常量中"
                    ))
                self.generic_visit(node)

        HardcodedVisitor().visit(tree)
        return issues

    def _check_long_functions(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查过长的函数"""
        issues = []

        class LongFunctionVisitor(ast.NodeVisitor):
            def visit_FunctionDef(self, node):
                # 计算函数行数
                if hasattr(node, 'end_lineno'):
                    function_lines = node.end_lineno - node.lineno + 1
                else:
                    function_lines = len(node.body) if node.body else 0

                if function_lines > 50:
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="long_function",
                        message=f"函数过长: {function_lines} 行",
                        severity="warning",
                        suggestion="考虑将函数拆分为更小的函数"
                    ))

                self.generic_visit(node)

        LongFunctionVisitor().visit(tree)
        return issues

    def _check_complex_functions(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查复杂的函数"""
        issues = []

        class ComplexFunctionVisitor(ast.NodeVisitor):
            def visit_FunctionDef(self, node):
                complexity = self._calculate_complexity(node)

                if complexity > 10:
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="complex_function",
                        message=f"函数复杂度过高: {complexity}",
                        severity="warning",
                        suggestion="考虑简化函数逻辑，减少条件分支"
                    ))

                self.generic_visit(node)

            def _calculate_complexity(self, node):
                """计算圈复杂度"""
                complexity = 1  # 基础复杂度

                for child in ast.walk(node):
                    if isinstance(child, (ast.If, ast.While, ast.For, ast.AsyncFor)):
                        complexity += 1
                    elif isinstance(child, ast.ExceptHandler):
                        complexity += 1
                    elif isinstance(child, ast.BoolOp):
                        complexity += len(child.values) - 1

                return complexity

        ComplexFunctionVisitor().visit(tree)
        return issues

    def _check_duplicate_code(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查重复代码"""
        issues = []

        # 简单的重复代码检测
        lines = content.split('\n')
        line_counts = {}

        for i, line in enumerate(lines):
            stripped_line = line.strip()
            if len(stripped_line) > 20:  # 只检查较长的行
                if stripped_line in line_counts:
                    line_counts[stripped_line].append(i + 1)
                else:
                    line_counts[stripped_line] = [i + 1]

        for line, line_numbers in line_counts.items():
            if len(line_numbers) > 2:  # 出现3次以上
                issues.append(CodeIssue(
                    file_path=str(file_path),
                    line_number=line_numbers[0],
                    issue_type="duplicate_code",
                    message=f"重复代码行，共出现 {len(line_numbers)} 次",
                    severity="info",
                    suggestion="考虑提取为函数或常量"
                ))

        return issues

    def _check_unused_imports(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查未使用的导入"""
        issues = []

        class UnusedImportVisitor(ast.NodeVisitor):
            def __init__(self):
                self.imports = set()
                self.used_names = set()

            def visit_Import(self, node):
                for alias in node.names:
                    self.imports.add(alias.name)
                self.generic_visit(node)

            def visit_ImportFrom(self, node):
                if node.module:
                    for alias in node.names:
                        self.imports.add(alias.name)
                self.generic_visit(node)

            def visit_Name(self, node):
                self.used_names.add(node.id)
                self.generic_visit(node)

            def visit_Attribute(self, node):
                if isinstance(node.value, ast.Name):
                    self.used_names.add(node.value.id)
                self.generic_visit(node)

        visitor = UnusedImportVisitor()
        visitor.visit(tree)

        unused_imports = visitor.imports - visitor.used_names
        for unused_import in unused_imports:
            issues.append(CodeIssue(
                file_path=str(file_path),
                line_number=0,  # 需要更精确的行号
                issue_type="unused_import",
                message=f"未使用的导入: {unused_import}",
                severity="info",
                suggestion="删除未使用的导入"
            ))

        return issues

    def _check_magic_numbers(self, tree: ast.AST, file_path: Path, content: str) -> List[CodeIssue]:
        """检查魔法数字"""
        issues = []

        class MagicNumberVisitor(ast.NodeVisitor):
            def visit_Num(self, node):
                if isinstance(node.n, int) and node.n not in [0, 1, -1]:
                    issues.append(CodeIssue(
                        file_path=str(file_path),
                        line_number=node.lineno,
                        issue_type="magic_number",
                        message=f"魔法数字: {node.n}",
                        severity="info",
                        suggestion="使用有意义的常量名"
                    ))
                self.generic_visit(node)

        MagicNumberVisitor().visit(tree)
        return issues

    def generate_report(self) -> str:
        """生成检查报告"""
        report = []
        report.append("=" * 80)
        report.append("VirtualChemLab 代码质量检查报告")
        report.append("=" * 80)
        report.append(f"检查时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append(f"总计问题: {len(self.issues)} 个")
        report.append("")

        # 按严重程度分组
        by_severity = {}
        for issue in self.issues:
            if issue.severity not in by_severity:
                by_severity[issue.severity] = []
            by_severity[issue.severity].append(issue)

        # 按严重程度排序
        severity_order = ["error", "warning", "info"]

        for severity in severity_order:
            if severity in by_severity:
                issues = by_severity[severity]
                report.append(f"## {severity.upper()} 级别 ({len(issues)} 个)")
                report.append("-" * 40)

                for issue in issues:
                    report.append(f"### {issue.file_path}:{issue.line_number}")
                    report.append(f"类型: {issue.issue_type}")
                    report.append(f"消息: {issue.message}")
                    if issue.suggestion:
                        report.append(f"建议: {issue.suggestion}")
                    report.append("")

        return "\n".join(report)
```
