## 概述

服务实现层提供了4个核心服务的具体实现：

| 服务 | 描述 | 主要功能 |
|------|------|----------|
| `ExperimentServiceImpl` | 实验服务 | 创建、执行、管理实验 |
| `StorageServiceImpl` | 存储服务 | 数据持久化、查询 |
| `ReportServiceImpl` | 报告服务 | 生成、导出实验报告 |
| `PluginServiceImpl` | 插件服务 | 管理、执行插件 |

所有服务实现都遵循**依赖注入**原则，可通过DI容器自动管理。

## ExperimentServiceImpl - 实验服务

### 创建服务实例

```python
from src.services import ExperimentServiceImpl
from src.contracts.experiment_service import ExperimentServiceConfig

# 手动创建
service = ExperimentServiceImpl(
    engine=experiment_engine,
    storage=storage,
    config=ExperimentServiceConfig(
        auto_save=True,
        enable_safety_check=True
    )
)

# 通过DI容器（推荐）
from src.core.service_registration import configure_container

container = configure_container()
service = container.resolve(ExperimentService)
```

### 使用示例

#### 1. 创建实验

```python
from src.contracts.experiment_service import ExperimentRequest

request = ExperimentRequest(
    user_id="user123",
    template_id="acid_base_titration",
    options={
        "difficulty": "beginner",
        "enable_hints": True
    }
)

response = service.create_experiment(request)

if response.success:
    print(f"✅ 实验创建成功")
    print(f"   实验ID: {response.experiment_id}")
    print(f"   状态: {response.status}")
else:
    print(f"❌ 创建失败: {response.message}")
```

#### 2. 开始实验

```python
experiment_id = "exp-123-456"
response = service.start_experiment(experiment_id)

if response.success:
    print(f"✅ 实验已开始")
    print(f"   当前状态: {response.status}")
```

#### 3. 提交步骤

```python
from src.contracts.experiment_service import StepSubmissionRequest

request = StepSubmissionRequest(
    experiment_id="exp-123-456",
    step_id="step-1",
    user_input={
        "reagent": "HCl",
        "volume": 25.0,
        "concentration": 0.1
    }
)

response = service.submit_step(request)

if response.passed:
    print(f"✅ 步骤通过")
    print(f"   反馈: {response.message}")
    if response.next_step_id:
        print(f"   下一步: {response.next_step_id}")
else:
    print(f"❌ 步骤错误")
    if response.mistake:
        print(f"   错误: {response.mistake.description}")
        print(f"   提示: {response.mistake.suggestion}")
```

#### 4. 获取进度

```python
progress = service.get_progress("exp-123-456")

if progress:
    print(f"实验进度:")
    print(f"  当前步骤: {progress.current_step}/{progress.total_steps}")
    print(f"  完成度: {progress.completion_rate:.1%}")
    print(f"  错误次数: {progress.total_mistakes}")
```

#### 5. 完成实验

```python
response = service.complete_experiment("exp-123-456")

if response.success:
    print(f"✅ 实验完成")
    print(f"   记录ID: {response.record_id}")
```

## StorageServiceImpl - 存储服务

### 创建服务实例

```python
from src.services import StorageServiceImpl
from src.storage.json_store import JSONStore

storage = JSONStore("data/")
service = StorageServiceImpl(
    storage=storage,
    config=StorageServiceConfig(
        enable_cache=True,
        max_cache_size=1000
    )
)
```

### 使用示例

#### 1. 保存实体

```python
from src.contracts.storage_service import SaveRequest

request = SaveRequest(
    entity_type="user",
    entity={"id": "user123", "name": "Alice", "email": "alice@example.com"},
    metadata={"source": "registration"}
)

response = service.save(request)

if response.success:
    print(f"✅ 保存成功: {response.entity_id}")
```

#### 2. 查询实体

```python
from src.contracts.storage_service import (
    QueryRequest,
    QueryFilter,
    QueryOperator
)

request = QueryRequest(
    entity_type="user",
    filters=[
        QueryFilter(field="age", operator=QueryOperator.GT, value=18),
        QueryFilter(field="status", operator=QueryOperator.EQ, value="active")
    ],
    sort_by="created_at",
    sort_order="desc",
    limit=10
)

response = service.query(request)

if response.success:
    print(f"✅ 找到 {response.total_count} 条记录")
    for entity in response.data:
        print(f"   - {entity}")
```

#### 3. 支持的查询操作符

| 操作符 | 说明 | 示例 |
|--------|------|------|
| `EQ` | 等于 | `age == 25` |
| `NE` | 不等于 | `status != "deleted"` |
| `GT` | 大于 | `score > 80` |
| `GTE` | 大于等于 | `score >= 60` |
| `LT` | 小于 | `price < 100` |
| `LTE` | 小于等于 | `age <= 65` |
| `IN` | 包含 | `role in ["admin", "editor"]` |
| `LIKE` | 模糊匹配 | `name like "Alice"` |

#### 4. 批量操作

```python
requests = [
    SaveRequest(entity_type="user", entity=user1),
    SaveRequest(entity_type="user", entity=user2),
    SaveRequest(entity_type="user", entity=user3),
]

responses = service.batch_save(requests)

success_count = sum(1 for r in responses if r.success)
print(f"✅ 成功保存 {success_count}/{len(requests)} 条记录")
```

## ReportServiceImpl - 报告服务

### 创建服务实例

```python
from src.services import ReportServiceImpl
from src.core.repository import FileRepository

# 带记录仓储（推荐）
record_repo = FileRepository(UserRecord, "data/records/")

service = ReportServiceImpl(
    generator=report_generator,
    exporter=report_exporter,
    record_repository=record_repo,  # 可选，但推荐
    config=ReportServiceConfig(output_dir="reports/")
)
```

### 使用示例

#### 1. 生成实验报告

```python
from src.contracts.report_service import ExportFormat

# 假设已有实验记录
record = user_records[0]

response = service.generate_experiment_report(
    record=record,
    format=ExportFormat.PDF,
    options={"include_charts": True, "language": "zh_CN"}
)

if response.success:
    print(f"✅ 报告生成成功")
    print(f"   报告ID: {response.report_id}")
    print(f"   文件路径: {response.file_path}")
```

#### 2. 生成汇总报告

```python
response = service.generate_summary_report(
    user_id="user123",
    start_date="2025-01-01",
    end_date="2025-12-31",
    format=ExportFormat.HTML
)

if response.success:
    print(f"✅ 汇总报告生成成功: {response.file_path}")
```

#### 3. 生成对比报告

```python
record_ids = ["rec-001", "rec-002", "rec-003"]

response = service.generate_comparison_report(
    record_ids=record_ids,
    format=ExportFormat.PDF
)

if response.success:
    print(f"✅ 对比报告生成成功: {response.file_path}")
```

#### 4. 支持的导出格式

| 格式 | 说明 | 文件扩展名 |
|------|------|-----------|
| `PDF` | PDF文档 | `.pdf` |
| `HTML` | HTML网页 | `.html` |
| `MARKDOWN` | Markdown文档 | `.md` |
| `JSON` | JSON数据 | `.json` |
| `DOCX` | Word文档 | `.docx` |
| `CSV` | CSV表格 | `.csv` |

#### 5. 报告预览

```python

    report_type=ReportType.EXPERIMENT,
    record_id="rec-123"

```

## PluginServiceImpl - 插件服务

### 创建服务实例

```python
from src.services import PluginServiceImpl

service = PluginServiceImpl(
    loader=plugin_loader,
    registry=plugin_registry,
    config=PluginServiceConfig(
        enable_sandboxing=True,
        timeout=30
    )
)
```

### 使用示例

#### 1. 执行插件

```python
from src.contracts.plugin_service import PluginExecuteRequest

request = PluginExecuteRequest(
    plugin_name="data_analyzer",
    action="analyze",
    params={
        "data": experiment_data,
        "metrics": ["mean", "std", "variance"]
    }
)

response = service.execute_plugin(request)

if response.success:
    print(f"✅ 插件执行成功")
    print(f"   结果: {response.result}")
else:
    print(f"❌ 执行失败: {response.error}")
```

#### 2. 列出插件

```python
from src.contracts.plugin_service import PluginType, PluginStatus

# 列出所有已启用的数据处理插件
plugins = service.list_plugins(
    plugin_type=PluginType.DATA_PROCESSOR,
    status=PluginStatus.ENABLED
)

for plugin in plugins:
    print(f"📦 {plugin.name} v{plugin.version}")
    print(f"   能力: {', '.join(plugin.capabilities)}")
```

#### 3. 安装插件

```python
from src.contracts.plugin_service import PluginInstallRequest

request = PluginInstallRequest(
    plugin_path="plugins/my_plugin.py",
    config={"api_key": "xxx"}
)

response = service.install_plugin(request)

if response.success:
    print(f"✅ 插件安装成功: {response.plugin_name}")
```

#### 4. 插件管理

```python
# 禁用插件
service.disable_plugin("old_plugin")

# 启用插件
service.enable_plugin("old_plugin")

# 卸载插件
service.uninstall_plugin("unused_plugin")
```

## 常见问题

### Q: 如何扩展服务功能？

A: 创建子类并重写方法：

```python
class CustomExperimentService(ExperimentServiceImpl):
    def create_experiment(self, request):
        # 添加自定义逻辑
        logger.info(f"自定义创建逻辑: {request}")

        # 调用父类方法
        response = super().create_experiment(request)

        # 后处理
        if response.success:
            self.notify_admins(response.experiment_id)

        return response
```

### Q: 如何切换不同的存储后端？

A: 实现 `IStorage` 接口并在DI容器中注册：

```python
from src.interfaces.storage import IStorage

class DatabaseStorage(IStorage):
    # 实现所有接口方法
    pass

container.register_singleton(IStorage, DatabaseStorage)
```

### Q: 服务之间如何通信？

A: 通过事件总线：

```python
from src.core.event_bus import Event, get_event_bus

# 发布事件
event_bus = get_event_bus()
event_bus.publish(Event(
    name="experiment.completed",
    data={"experiment_id": "exp-123"}
))

# 订阅事件
def on_experiment_completed(event):
    # 触发报告生成
    report_service.generate_experiment_report(...)

event_bus.subscribe("experiment.completed", on_experiment_completed)
```
