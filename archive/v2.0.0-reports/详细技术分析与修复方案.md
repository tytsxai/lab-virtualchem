**性能问题**:

- 时间复杂度为O(n)，缓存越大性能越差
- 缺乏高效的数据结构支持

#### 优化方案

```python
from collections import OrderedDict
import heapq
from typing import Dict, Any

class OptimizedLRUCache:
    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self.cache: OrderedDict[str, Any] = OrderedDict()
        self.access_times: Dict[str, float] = {}
        self._access_counter = 0

    def get(self, key: str) -> Any:
        if key in self.cache:
            # 移动到末尾（最近使用）
            self.cache.move_to_end(key)
            self.access_times[key] = self._access_counter
            self._access_counter += 1
            return self.cache[key]
        return None

    def set(self, key: str, value: Any) -> None:
        if key in self.cache:
            # 更新现有项
            self.cache[key] = value
            self.cache.move_to_end(key)
        else:
            # 添加新项
            if len(self.cache) >= self.max_size:
                # 移除最旧的项（O(1)操作）
                self.cache.popitem(last=False)
            self.cache[key] = value
        self.access_times[key] = self._access_counter
        self._access_counter += 1
```

#### 当前问题


**潜在问题**:

- 虽然使用了双重检查锁，但容器内部操作可能不是线程安全的
- 服务解析过程中的依赖注入可能存在竞态条件
- 缺乏对容器状态的保护

#### 修复方案

```python
import threading
from typing import Dict, Any, TypeVar, Generic

T = TypeVar('T')

class ThreadSafeDIContainer:
    def __init__(self):
        self._services: Dict[type, Any] = {}
        self._singletons: Dict[type, Any] = {}
        self._lock = threading.RLock()  # 使用可重入锁
        self._resolving: set = set()

    def resolve(self, service_type: type[T]) -> T:
        with self._lock:
            # 检查循环依赖
            if service_type in self._resolving:
                raise RuntimeError(f"Circular dependency detected: {service_type}")

            # 检查是否已注册
            if service_type not in self._services:
                raise ValueError(f"Service {service_type} not registered")

            # 如果是单例且已创建，直接返回
            if service_type in self._singletons:
                return self._singletons[service_type]

            # 标记正在解析
            self._resolving.add(service_type)
            try:
                # 创建实例
                instance = self._create_instance(service_type)

                # 如果是单例，缓存实例
                if self._services[service_type].lifetime == Lifetime.SINGLETON:
                    self._singletons[service_type] = instance

                return instance
            finally:
                self._resolving.remove(service_type)
```

```python
import threading
from typing import Dict, Any, Callable
from contextlib import contextmanager

class ThreadSafeManager:
    """全局线程安全管理器"""

    def __init__(self):
        self._locks: Dict[str, threading.RLock] = {}
        self._global_lock = threading.RLock()

    def get_lock(self, name: str) -> threading.RLock:
        """获取命名锁"""
        with self._global_lock:
            if name not in self._locks:
                self._locks[name] = threading.RLock()
            return self._locks[name]

    @contextmanager
    def locked(self, name: str):
        """上下文管理器，自动获取和释放锁"""
        lock = self.get_lock(name)
        lock.acquire()
        try:
            yield
        finally:
            lock.release()
```

```python
from collections import OrderedDict
from typing import Dict, Any, Optional
import time

class OptimizedLRUCache:
    """优化的LRU缓存实现"""

    def __init__(self, max_size: int = 1000, ttl: int = 3600):
        self.max_size = max_size
        self.ttl = ttl
        self.cache: OrderedDict[str, Dict[str, Any]] = OrderedDict()
        self.stats = {"hits": 0, "misses": 0, "evictions": 0}

    def get(self, key: str) -> Optional[Any]:
        if key in self.cache:
            entry = self.cache[key]

            # 检查是否过期
            if time.time() - entry["created_at"] > self.ttl:
                del self.cache[key]
                self.stats["misses"] += 1
                return None

            # 移动到末尾（最近使用）
            self.cache.move_to_end(key)
            entry["last_accessed"] = time.time()
            entry["access_count"] += 1
            self.stats["hits"] += 1
            return entry["value"]

        self.stats["misses"] += 1
        return None

    def set(self, key: str, value: Any) -> None:
        current_time = time.time()

        if key in self.cache:
            # 更新现有项
            self.cache[key]["value"] = value
            self.cache[key]["last_accessed"] = current_time
            self.cache.move_to_end(key)
        else:
            # 添加新项
            if len(self.cache) >= self.max_size:
                # 移除最旧的项
                self.cache.popitem(last=False)
                self.stats["evictions"] += 1

            self.cache[key] = {
                "value": value,
                "created_at": current_time,
                "last_accessed": current_time,
                "access_count": 1
            }

    def warm_up(self, data: Dict[str, Any]) -> None:
        """预热缓存"""
        for key, value in data.items():
            self.set(key, value)

    def get_stats(self) -> Dict[str, Any]:
        total_requests = self.stats["hits"] + self.stats["misses"]
        hit_rate = self.stats["hits"] / total_requests if total_requests > 0 else 0

        return {
            **self.stats,
            "hit_rate": hit_rate,
            "size": len(self.cache),
            "max_size": self.max_size
        }
```

```python
import re
from functools import lru_cache
from typing import Dict, List, Callable, Any
from collections import defaultdict

class OptimizedEventBus:
    """优化的事件总线"""

    def __init__(self, max_history: int = 500):
        self._subscribers: List[EventSubscriber] = []
        self._history: List[Event] = []
        self._max_history = max_history

        # 性能优化：缓存编译的正则表达式
        self._pattern_cache: Dict[str, re.Pattern] = {}

        # 性能优化：缓存订阅者匹配结果
        self._subscriber_cache: Dict[str, List[EventSubscriber]] = defaultdict(list)

    @lru_cache(maxsize=1000)
    def _compile_pattern(self, pattern: str) -> Optional[re.Pattern]:
        """缓存编译的正则表达式"""
        if "*" in pattern:
            regex = pattern.replace(".", r"\.").replace("*", ".*")
            return re.compile(f"^{regex}$")
        return None

    def _match_pattern_cached(self, event_name: str, pattern: str) -> bool:
        """使用缓存的正则表达式进行匹配"""
        compiled_pattern = self._compile_pattern(pattern)
        if compiled_pattern:
            return bool(compiled_pattern.match(event_name))
        return event_name == pattern

    def subscribe(self, event_name: str, handler: Callable, priority: EventPriority = EventPriority.NORMAL) -> "OptimizedEventBus":
        """订阅事件（优化版本）"""
        subscriber = EventSubscriber(
            handler=handler,
            event_name=event_name,
            priority=priority
        )

        self._subscribers.append(subscriber)

        # 更新缓存
        self._subscriber_cache[event_name].append(subscriber)

        # 按优先级排序
        self._subscribers.sort(key=lambda s: s.priority.value, reverse=True)

        return self

    def publish(self, event: Event) -> List[Any]:
        """发布事件（优化版本）"""
        # 添加到历史
        self._add_to_history(event)

        # 获取匹配的订阅者（使用缓存）
        matching_subscribers = self._get_matching_subscribers(event.name)

        # 通知订阅者
        results = []
        for subscriber in matching_subscribers:
            if subscriber.matches(event):
                result = subscriber.handler(event)
                results.append(result)

        return results

    def _get_matching_subscribers(self, event_name: str) -> List[EventSubscriber]:
        """获取匹配的订阅者（使用缓存优化）"""
        # 首先检查精确匹配
        if event_name in self._subscriber_cache:
            return self._subscriber_cache[event_name]

        # 然后检查模式匹配
        matching_subscribers = []
        for subscriber in self._subscribers:
            if self._match_pattern_cached(event_name, subscriber.event_name):
                matching_subscribers.append(subscriber)

        return matching_subscribers
```

```python
from typing import Any, Callable, TypeVar, Optional
import logging
from functools import wraps

T = TypeVar('T')

class CommonErrorHandler:
    """公共错误处理器"""

    @staticmethod
    def handle_common_errors(func: Callable[..., T]) -> Callable[..., Optional[T]]:
        """通用错误处理装饰器"""
        @wraps(func)
        def wrapper(*args, **kwargs) -> Optional[T]:
            try:
                return func(*args, **kwargs)
            except ValueError as e:
                logging.error(f"值错误 in {func.__name__}: {e}")
                return None
            except TypeError as e:
                logging.error(f"类型错误 in {func.__name__}: {e}")
                return None
            except Exception as e:
                logging.error(f"未知错误 in {func.__name__}: {e}")
                return None
        return wrapper

    @staticmethod
    def safe_execute(func: Callable[..., T], *args, **kwargs) -> Optional[T]:
        """安全执行函数"""
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logging.error(f"执行失败: {e}")
            return None

class UnifiedLogger:
```

```python
# 在 tests/performance_benchmarks.py 中创建
import time
import pytest
from typing import Dict, Any
import statistics

class PerformanceBenchmark:
    """性能基准测试框架"""

    def __init__(self):
        self.results: Dict[str, List[float]] = {}

    def benchmark(self, name: str, func: Callable, *args, **kwargs) -> Dict[str, float]:
        """执行基准测试"""
        times = []

        # 预热
        for _ in range(10):
            func(*args, **kwargs)

        # 正式测试
        for _ in range(100):
            start = time.perf_counter()
            func(*args, **kwargs)
            end = time.perf_counter()
            times.append(end - start)

        # 计算统计信息
        stats = {
            "mean": statistics.mean(times),
            "median": statistics.median(times),
            "min": min(times),
            "max": max(times),
            "std": statistics.stdev(times) if len(times) > 1 else 0
        }

        self.results[name] = times
        return stats

    def compare(self, baseline: str, current: str) -> Dict[str, float]:
        """比较两个基准测试结果"""
        if baseline not in self.results or current not in self.results:
            raise ValueError("基准测试结果不存在")

        baseline_mean = statistics.mean(self.results[baseline])
        current_mean = statistics.mean(self.results[current])

        improvement = (baseline_mean - current_mean) / baseline_mean * 100

        return {
            "baseline_mean": baseline_mean,
            "current_mean": current_mean,
            "improvement_percent": improvement,
            "is_improved": improvement > 0
        }

# 使用示例
```

```python
        )

    def get_all_metrics(self) -> List[HealthMetric]:
        """获取所有健康指标"""
        metrics = [
            self.check_memory_usage(),
            self.check_cpu_usage(),
            self.check_disk_usage()
        ]

        # 检查告警
        for metric in metrics:
            if metric.status == HealthStatus.CRITICAL:
                self.alerts.append(f"严重告警: {metric.name} = {metric.value}{metric.unit}")
            elif metric.status == HealthStatus.WARNING:
                self.alerts.append(f"警告: {metric.name} = {metric.value}{metric.unit}")

        return metrics

    def get_health_summary(self) -> Dict[str, Any]:
        """获取健康摘要"""
        metrics = self.get_all_metrics()

        critical_count = sum(1 for m in metrics if m.status == HealthStatus.CRITICAL)
        warning_count = sum(1 for m in metrics if m.status == HealthStatus.WARNING)

        overall_status = HealthStatus.CRITICAL if critical_count > 0 else (
            HealthStatus.WARNING if warning_count > 0 else HealthStatus.HEALTHY
        )

        return {
            "overall_status": overall_status.value,
            "critical_count": critical_count,
            "warning_count": warning_count,
            "alerts": self.alerts,
            "timestamp": time.time()
        }
```
