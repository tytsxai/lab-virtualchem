```python
import ast
import re
from typing import Any, List

class AdvancedCodeInjectionDetector:
    """高级代码注入检测器"""

    def __init__(self):
        self.dangerous_patterns = [
            # 直接执行
            r'\beval\s*\(',
            r'\bexec\s*\(',
            r'\bcompile\s*\(',
            r'\b__import__\s*\(',

            # 系统调用
            r'\bos\.system\s*\(',
            r'\bsubprocess\s*\.',
            r'\bcommands\s*\.',
            r'\bpopen\s*\(',

            # 文件操作
            r'\bopen\s*\(',
            r'\bfile\s*\(',
            r'\binput\s*\(',
            r'\braw_input\s*\(',

            # 反射操作
            r'\bgetattr\s*\(',
            r'\bsetattr\s*\(',
            r'\bdelattr\s*\(',
            r'\bhasattr\s*\(',
            r'\bglobals\s*\(',
            r'\blocals\s*\(',

            # 危险属性访问
            r'\.__\w+__',
            r'\.__class__',
            r'\.__bases__',
            r'\.__subclasses__',
        ]

        self.compiled_patterns = [
            re.compile(pattern, re.IGNORECASE)
            for pattern in self.dangerous_patterns
        ]

    def detect_code_injection(self, code: str) -> List[str]:
        """检测代码注入"""
        threats = []

        # 1. 正则表达式检测
        for pattern in self.compiled_patterns:
            if pattern.search(code):
                threats.append(f"检测到危险模式: {pattern.pattern}")

        # 2. AST解析检测
        try:
            tree = ast.parse(code)
            threats.extend(self._analyze_ast(tree))
        except SyntaxError:
            threats.append("语法错误，可能存在注入")

        # 3. 编码变体检测
        threats.extend(self._detect_encoding_variants(code))

        return threats

    def _analyze_ast(self, node: ast.AST) -> List[str]:
        """分析AST节点"""
        threats = []

        class DangerousVisitor(ast.NodeVisitor):
            def visit_Call(self, node):
                # 检查危险函数调用
                if isinstance(node.func, ast.Name):
                    if node.func.id in ['eval', 'exec', 'compile', '__import__']:
                        threats.append(f"危险函数调用: {node.func.id}")

                # 检查属性调用
                if isinstance(node.func, ast.Attribute):
                    if node.func.attr in ['system', 'popen', 'call']:
                        threats.append(f"危险方法调用: {node.func.attr}")

                self.generic_visit(node)

            def visit_Attribute(self, node):
                # 检查危险属性访问
                if node.attr.startswith('__') and node.attr.endswith('__'):
                    threats.append(f"危险属性访问: {node.attr}")

                self.generic_visit(node)

        visitor = DangerousVisitor()
        visitor.visit(node)
        return threats

    def _detect_encoding_variants(self, code: str) -> List[str]:
        """检测编码变体"""
        threats = []

        # Unicode转义
        unicode_patterns = [
            r'\\u[0-9a-fA-F]{4}',
            r'\\x[0-9a-fA-F]{2}',
            r'\\[0-7]{1,3}',
        ]

        for pattern in unicode_patterns:
            if re.search(pattern, code):
                threats.append(f"检测到Unicode转义: {pattern}")

        # Base64编码检测
        import base64
        try:
            # 尝试解码可能的Base64
            if len(code) % 4 == 0 and re.match(r'^[A-Za-z0-9+/]*={0,2}$', code):
                decoded = base64.b64decode(code).decode('utf-8', errors='ignore')
                if any(keyword in decoded.lower() for keyword in ['eval', 'exec', 'system']):
                    threats.append("检测到Base64编码的恶意代码")
        except:
            pass

        return threats
```

**问题描述**:

- 旧密码格式使用不安全的SHA256

**修复方案**:

```python
import bcrypt
import time
from typing import Dict, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta

@dataclass
class LoginAttempt:
    """登录尝试记录"""
    username: str
    timestamp: datetime
    ip_address: str
    success: bool

class SecurePasswordManager:
    """安全密码管理器"""

    def __init__(self):
        self.login_attempts: Dict[str, List[LoginAttempt]] = {}
        self.max_attempts = 5
        self.lockout_duration = timedelta(minutes=15)

    def hash_password(self, password: str) -> str:
        """安全哈希密码"""
        # 验证密码强度
        self._validate_password_strength(password)

        # 使用bcrypt
        salt = bcrypt.gensalt(rounds=12)
        password_hash = bcrypt.hashpw(password.encode('utf-8'), salt)
        return password_hash.decode('utf-8')

    def verify_password(self, password: str, password_hash: str) -> bool:
        """验证密码"""
        try:
            return bcrypt.checkpw(password.encode('utf-8'), password_hash.encode('utf-8'))
        except Exception:
            return False

    def _validate_password_strength(self, password: str) -> None:
        """验证密码强度"""
        if len(password) < 8:
            raise ValueError("密码长度至少8位")

        if not re.search(r'[A-Z]', password):
            raise ValueError("密码必须包含大写字母")

        if not re.search(r'[a-z]', password):
            raise ValueError("密码必须包含小写字母")

        if not re.search(r'\d', password):
            raise ValueError("密码必须包含数字")

        if not re.search(r'[!@#$%^&*(),.?":{}|<>]', password):
            raise ValueError("密码必须包含特殊字符")

    def record_login_attempt(self, username: str, ip_address: str, success: bool) -> None:
        """记录登录尝试"""
        attempt = LoginAttempt(
            username=username,
            timestamp=datetime.now(),
            ip_address=ip_address,
            success=success
        )

        if username not in self.login_attempts:
            self.login_attempts[username] = []

        self.login_attempts[username].append(attempt)

        # 清理过期记录
        self._cleanup_old_attempts(username)

    def is_account_locked(self, username: str) -> bool:
        """检查账户是否被锁定"""
        if username not in self.login_attempts:
            return False

        attempts = self.login_attempts[username]
        recent_attempts = [
            a for a in attempts
            if datetime.now() - a.timestamp < self.lockout_duration
        ]

        failed_attempts = [a for a in recent_attempts if not a.success]
        return len(failed_attempts) >= self.max_attempts

    def _cleanup_old_attempts(self, username: str) -> None:
        """清理过期登录记录"""
        cutoff_time = datetime.now() - self.lockout_duration
        self.login_attempts[username] = [
            a for a in self.login_attempts[username]
            if a.timestamp > cutoff_time
        ]
```

```python
import uuid
from typing import Set, Dict
from datetime import datetime, timedelta

class SecureJWTManager:
    """安全JWT管理器"""

    def __init__(self, secret_key: str, algorithm: str = "HS256"):
        self.secret_key = secret_key
        self.algorithm = algorithm
        self.revoked_tokens: Set[str] = set()
        self.token_blacklist: Dict[str, datetime] = {}

    def create_access_token(self, user_id: str, jti: str = None) -> str:
        """创建访问令牌"""
        now = datetime.now()  # 使用本地时间
        expires = now + timedelta(hours=1)

        # 生成唯一令牌ID
        if jti is None:
            jti = str(uuid.uuid4())

        payload = {
            "sub": user_id,
            "iat": now,
            "exp": expires,
            "type": "access",
            "jti": jti,  # JWT ID
            "iss": "virtualchemlab",  # 发行者
            "aud": "virtualchemlab-users"  # 受众
        }

        return jwt.encode(payload, self.secret_key, algorithm=self.algorithm)

    def verify_token(self, token: str) -> Optional[Dict]:
        """验证令牌"""
        try:
            # 检查是否被撤销
            payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm])

            jti = payload.get("jti")
            if jti and jti in self.revoked_tokens:
                raise jwt.InvalidTokenError("令牌已被撤销")

            # 检查黑名单
            if jti and jti in self.token_blacklist:
                if datetime.now() < self.token_blacklist[jti]:
                    raise jwt.InvalidTokenError("令牌在黑名单中")

            return payload

        except jwt.ExpiredSignatureError:
            raise jwt.InvalidTokenError("令牌已过期")
        except jwt.InvalidTokenError:
            raise jwt.InvalidTokenError("无效的令牌")

    def revoke_token(self, token: str) -> bool:
        """撤销令牌"""
        try:
            payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm])
            jti = payload.get("jti")

            if jti:
                self.revoked_tokens.add(jti)
                # 设置黑名单过期时间
                self.token_blacklist[jti] = datetime.now() + timedelta(hours=24)
                return True

            return False

        except jwt.InvalidTokenError:
            return False

    def cleanup_expired_tokens(self) -> None:
        """清理过期令牌"""
        now = datetime.now()
        expired_jtis = [
            jti for jti, expiry in self.token_blacklist.items()
            if now > expiry
        ]

        for jti in expired_jtis:
            self.revoked_tokens.discard(jti)
            del self.token_blacklist[jti]
```

```python
import magic
from PIL import Image
import zipfile
import xml.etree.ElementTree as ET

class AdvancedInputValidator:
    """高级输入验证器"""

    def __init__(self):
        self.max_file_size = 10 * 1024 * 1024  # 10MB
        self.allowed_mime_types = {
            'image/jpeg', 'image/png', 'image/gif',
            'application/pdf', 'text/plain',
            'application/json', 'text/csv'
        }
        self.dangerous_extensions = {
            '.exe', '.bat', '.cmd', '.scr', '.pif',
            '.com', '.vbs', '.js', '.jar', '.class'
        }

    def validate_file_upload(self, file_path: str, file_content: bytes) -> Dict[str, Any]:
        """验证文件上传"""
        result = {
            "valid": False,
            "errors": [],
            "file_type": None,
            "file_size": len(file_content)
        }

        # 1. 检查文件大小
        if len(file_content) > self.max_file_size:
            result["errors"].append("文件大小超过限制")
            return result

        # 2. 检查文件扩展名
        file_ext = Path(file_path).suffix.lower()
        if file_ext in self.dangerous_extensions:
            result["errors"].append("不允许的文件类型")
            return result

        # 3. 检查MIME类型
        try:
            mime_type = magic.from_buffer(file_content, mime=True)
            result["file_type"] = mime_type

            if mime_type not in self.allowed_mime_types:
                result["errors"].append("不允许的MIME类型")
                return result

        except Exception as e:
            result["errors"].append(f"MIME类型检测失败: {e}")
            return result

        # 4. 特殊文件类型验证
        if mime_type.startswith('image/'):
            if not self._validate_image_file(file_content):
                result["errors"].append("无效的图片文件")
                return result

        elif mime_type == 'application/zip':
            if not self._validate_zip_file(file_content):
                result["errors"].append("无效的ZIP文件")
                return result

        elif mime_type == 'application/xml':
            if not self._validate_xml_file(file_content):
                result["errors"].append("无效的XML文件")
                return result

        result["valid"] = True
        return result

    def _validate_image_file(self, content: bytes) -> bool:
        """验证图片文件"""
        try:
            from io import BytesIO
            img = Image.open(BytesIO(content))
            img.verify()
            return True
        except Exception:
            return False

    def _validate_zip_file(self, content: bytes) -> bool:
        """验证ZIP文件"""
        try:
            with zipfile.ZipFile(BytesIO(content), 'r') as zip_file:
                # 检查是否有恶意文件
                for file_info in zip_file.filelist:
                    if file_info.filename.endswith(self.dangerous_extensions):
                        return False

                    # 检查路径遍历
                    if '..' in file_info.filename or file_info.filename.startswith('/'):
                        return False

                return True
        except Exception:
            return False

    def _validate_xml_file(self, content: bytes) -> bool:
        """验证XML文件"""
        try:
            # 检查XXE攻击
            if b'<!DOCTYPE' in content or b'<!ENTITY' in content:
                return False

            # 解析XML
            root = ET.fromstring(content)

            # 检查深度
            if self._get_xml_depth(root) > 10:
                return False

            return True
        except Exception:
            return False

    def _get_xml_depth(self, element, current_depth=0) -> int:
        """获取XML深度"""
        if not element:
            return current_depth

        max_depth = current_depth
        for child in element:
            depth = self._get_xml_depth(child, current_depth + 1)
            max_depth = max(max_depth, depth)

        return max_depth

    def validate_json_input(self, json_data: Any) -> Dict[str, Any]:
        """验证JSON输入"""
        result = {"valid": True, "errors": []}

        if isinstance(json_data, dict):
            # 检查键名
            for key in json_data.keys():
                if not isinstance(key, str):
                    result["errors"].append("JSON键必须是字符串")
                    result["valid"] = False

                if len(key) > 100:
                    result["errors"].append("JSON键名过长")
                    result["valid"] = False

                # 递归验证值
                value_result = self.validate_json_input(json_data[key])
                if not value_result["valid"]:
                    result["errors"].extend(value_result["errors"])
                    result["valid"] = False

        elif isinstance(json_data, list):
            # 检查列表长度
            if len(json_data) > 1000:
                result["errors"].append("JSON数组过长")
                result["valid"] = False

            # 递归验证元素
            for item in json_data:
                item_result = self.validate_json_input(item)
                if not item_result["valid"]:
                    result["errors"].extend(item_result["errors"])
                    result["valid"] = False

        elif isinstance(json_data, str):
            # 检查字符串长度
            if len(json_data) > 10000:
                result["errors"].append("JSON字符串过长")
                result["valid"] = False

        return result
```

```python
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64

class SecureLicenseManager:
    """安全许可证管理器"""

    def __init__(self, secret_key: str, license_file: Path):
        self.secret_key = secret_key
        self.license_file = license_file
        self._setup_encryption()

    def _setup_encryption(self):
        """设置加密"""
        # 从密钥派生加密密钥
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=b'virtualchemlab_license_salt',
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(self.secret_key.encode()))
        self.cipher = Fernet(key)

    def encrypt_license(self, license_obj: License) -> dict[str, Any]:
        """加密许可证"""
        data = license_obj.to_dict()

        # 序列化并加密
        json_data = json.dumps(data, sort_keys=True)
        encrypted_data = self.cipher.encrypt(json_data.encode())

        # 添加完整性校验
        import hmac
        mac = hmac.new(
            self.secret_key.encode(),
            encrypted_data,
            hashlib.sha256
        ).hexdigest()

        return {
            "data": base64.b64encode(encrypted_data).decode(),
            "mac": mac,
            "version": "2.0"
        }

    def decrypt_license(self, encrypted_data: dict[str, Any]) -> License:
        """解密许可证"""
        if encrypted_data.get("version") != "2.0":
            raise ValueError("不支持的许可证版本")

        # 验证完整性
        data_bytes = base64.b64decode(encrypted_data["data"])
        expected_mac = hmac.new(
            self.secret_key.encode(),
            data_bytes,
            hashlib.sha256
        ).hexdigest()

        if not hmac.compare_digest(expected_mac, encrypted_data["mac"]):
            raise ValueError("许可证数据被篡改")

        # 解密
        decrypted_data = self.cipher.decrypt(data_bytes)
        data = json.loads(decrypted_data.decode())

        return License.from_dict(data)

    def generate_secure_machine_id(self) -> str:
        """生成安全的机器ID"""
        import platform
        import psutil
        import uuid

        # 收集硬件信息
        components = [
            platform.node(),  # 主机名
            platform.machine(),  # 架构
            str(uuid.getnode()),  # MAC地址
        ]

        # 获取CPU信息
        try:
            cpu_info = platform.processor()
            components.append(cpu_info)
        except:
            pass

        # 获取主板信息
        try:
            import wmi
            w = wmi.WMI()
            for board in w.Win32_BaseBoard():
                components.append(board.SerialNumber)
                break
        except:
            pass

        # 生成唯一ID
        machine_string = ":".join(components)
        machine_hash = hashlib.sha256(machine_string.encode()).hexdigest()

        return machine_hash[:32]
```

```python
import re
from typing import Any, Dict

class LogSanitizer:
    """日志脱敏器"""

    def __init__(self):
        self.sensitive_patterns = [
            (r'password["\']?\s*[:=]\s*["\']?([^"\'\\s]+)', r'password="***"'),
            (r'token["\']?\s*[:=]\s*["\']?([^"\'\\s]+)', r'token="***"'),
            (r'key["\']?\s*[:=]\s*["\']?([^"\'\\s]+)', r'key="***"'),
            (r'secret["\']?\s*[:=]\s*["\']?([^"\'\\s]+)', r'secret="***"'),
            (r'email["\']?\s*[:=]\s*["\']?([^"\'\\s@]+@[^"\'\\s]+)', r'email="***@***"'),
            (r'phone["\']?\s*[:=]\s*["\']?(\d{3,})', r'phone="***"'),
        ]

        self.compiled_patterns = [
            (re.compile(pattern, re.IGNORECASE), replacement)
            for pattern, replacement in self.sensitive_patterns
        ]

    def sanitize_log_message(self, message: str) -> str:
        """脱敏日志消息"""
        sanitized = message

        for pattern, replacement in self.compiled_patterns:
            sanitized = pattern.sub(replacement, sanitized)

        return sanitized

    def sanitize_dict(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """脱敏字典数据"""
        sensitive_keys = {
            'password', 'token', 'key', 'secret', 'api_key',
            'access_token', 'refresh_token', 'private_key'
        }

        sanitized = {}
        for key, value in data.items():
            if key.lower() in sensitive_keys:
                sanitized[key] = "***"
            elif isinstance(value, dict):
                sanitized[key] = self.sanitize_dict(value)
            elif isinstance(value, str):
                sanitized[key] = self.sanitize_log_message(value)
            else:
                sanitized[key] = value

        return sanitized
```

```python
from typing import Callable, Any
import time
from functools import wraps

class SecurityMiddleware:
    """安全中间件"""

    def __init__(self):
        self.rate_limiter = RateLimiter()
        self.input_validator = AdvancedInputValidator()
        self.log_sanitizer = LogSanitizer()

    def secure_endpoint(self,
                       rate_limit: int = 100,
                       require_auth: bool = True,
                       validate_input: bool = True):
        """安全端点装饰器"""
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            def wrapper(*args, **kwargs):
                # 1. 速率限制
                if not self.rate_limiter.check_rate_limit(
                    client_ip=kwargs.get('client_ip', 'unknown'),
                    limit=rate_limit
                ):
                    raise SecurityError("请求过于频繁")

                # 2. 认证检查
                if require_auth:
                    token = kwargs.get('token')
                    if not token or not self.verify_token(token):
                        raise SecurityError("认证失败")

                # 3. 输入验证
                if validate_input:
                    for key, value in kwargs.items():
                        if isinstance(value, str):
                            self.input_validator.validate_text_input(value)

                # 4. 执行函数
                try:
                    result = func(*args, **kwargs)
                    return result
                except Exception as e:
                    # 5. 安全日志记录
                    self.log_security_event(func.__name__, str(e), kwargs)
                    raise

            return wrapper
        return decorator

    def log_security_event(self, function_name: str, error: str, context: Dict[str, Any]):
        """记录安全事件"""
        sanitized_context = self.log_sanitizer.sanitize_dict(context)

        security_log = {
            "timestamp": datetime.now().isoformat(),
            "event": "security_error",
            "function": function_name,
            "error": error,
            "context": sanitized_context
        }

        logger.warning(f"安全事件: {json.dumps(security_log)}")

class RateLimiter:
    """速率限制器"""

    def __init__(self):
        self.requests: Dict[str, List[float]] = {}
        self.cleanup_interval = 300  # 5分钟清理一次
        self.last_cleanup = time.time()

    def check_rate_limit(self, client_ip: str, limit: int = 100) -> bool:
        """检查速率限制"""
        now = time.time()

        # 清理过期记录
        if now - self.last_cleanup > self.cleanup_interval:
            self._cleanup_expired_requests(now)
            self.last_cleanup = now

        # 获取客户端请求记录
        if client_ip not in self.requests:
            self.requests[client_ip] = []

        requests = self.requests[client_ip]

        # 移除1分钟前的请求
        cutoff_time = now - 60
        requests[:] = [req_time for req_time in requests if req_time > cutoff_time]

        # 检查是否超过限制
        if len(requests) >= limit:
            return False

        # 记录当前请求
        requests.append(now)
        return True

    def _cleanup_expired_requests(self, now: float):
        """清理过期请求记录"""
        cutoff_time = now - 300  # 5分钟
        for client_ip in list(self.requests.keys()):
            self.requests[client_ip] = [
                req_time for req_time in self.requests[client_ip]
                if req_time > cutoff_time
            ]
            if not self.requests[client_ip]:
                del self.requests[client_ip]
```

```python
class SecurityMonitor:
    """安全监控器"""

    def __init__(self):
        self.suspicious_activities = []
        self.blocked_ips = set()
        self.alert_thresholds = {
            "failed_logins": 10,
            "suspicious_requests": 50,
            "rate_limit_violations": 20
        }

    def monitor_login_attempts(self, username: str, success: bool, ip: str):
        """监控登录尝试"""
        if not success:
            # 记录失败登录
            self._record_failed_login(username, ip)

            # 检查是否需要阻止IP
            if self._should_block_ip(ip):
                self.blocked_ips.add(ip)
                self._send_security_alert(f"IP {ip} 已被阻止")

    def monitor_api_requests(self, endpoint: str, ip: str, response_code: int):
        """监控API请求"""
        if response_code == 403:
            self._record_suspicious_request(endpoint, ip)

        if response_code == 429:
            self._record_rate_limit_violation(ip)

    def _should_block_ip(self, ip: str) -> bool:
        """检查是否应该阻止IP"""
        failed_count = sum(
            1 for activity in self.suspicious_activities
            if activity.get('ip') == ip and activity.get('type') == 'failed_login'
        )
        return failed_count >= self.alert_thresholds["failed_logins"]

    def _send_security_alert(self, message: str):
        """发送安全告警"""
        alert = {
            "timestamp": datetime.now().isoformat(),
            "level": "HIGH",
            "message": message,
            "action_required": True
        }

        logger.critical(f"安全告警: {json.dumps(alert)}")

        # 发送到外部监控系统
        self._send_to_monitoring_system(alert)
```
