```python
# 可维护性评估工具
from dataclasses import dataclass
from typing import List, Dict, Any
from pathlib import Path
import ast
import os

@dataclass
class MaintainabilityMetrics:
    """可维护性指标"""
    cyclomatic_complexity: float
    code_duplication: float
    coupling_coefficient: float
    cohesion_score: float
    documentation_coverage: float
    test_coverage: float
    technical_debt: float

class MaintainabilityAnalyzer:
    """可维护性分析器"""

    def __init__(self, project_path: Path):
        self.project_path = project_path
        self.metrics = MaintainabilityMetrics(
            cyclomatic_complexity=0.0,
            code_duplication=0.0,
            coupling_coefficient=0.0,
            cohesion_score=0.0,
            documentation_coverage=0.0,
            test_coverage=0.0,
            technical_debt=0.0
        )

    def analyze_project(self) -> MaintainabilityMetrics:
        """分析项目可维护性"""
        # 1. 分析圈复杂度
        self.metrics.cyclomatic_complexity = self._analyze_cyclomatic_complexity()

        # 2. 分析代码重复
        self.metrics.code_duplication = self._analyze_code_duplication()

        # 3. 分析耦合度
        self.metrics.coupling_coefficient = self._analyze_coupling()

        # 4. 分析内聚性
        self.metrics.cohesion_score = self._analyze_cohesion()

        # 5. 分析文档覆盖率
        self.metrics.documentation_coverage = self._analyze_documentation()

        # 6. 分析测试覆盖率
        self.metrics.test_coverage = self._analyze_test_coverage()

        # 7. 分析技术债务
        self.metrics.technical_debt = self._analyze_technical_debt()

        return self.metrics

    def _analyze_cyclomatic_complexity(self) -> float:
        """分析圈复杂度"""
        total_complexity = 0
        total_functions = 0

        for py_file in self.project_path.rglob("*.py"):
            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                tree = ast.parse(content)

                class ComplexityVisitor(ast.NodeVisitor):
                    def __init__(self):
                        self.complexity = 1
                        self.function_count = 0

                    def visit_FunctionDef(self, node):
                        self.function_count += 1
                        self._calculate_complexity(node)
                        self.generic_visit(node)

                    def _calculate_complexity(self, node):
                        for child in ast.walk(node):
                            if isinstance(child, (ast.If, ast.While, ast.For, ast.AsyncFor)):
                                self.complexity += 1
                            elif isinstance(child, ast.ExceptHandler):
                                self.complexity += 1
                            elif isinstance(child, ast.BoolOp):
                                self.complexity += len(child.values) - 1

                visitor = ComplexityVisitor()
                visitor.visit(tree)

                total_complexity += visitor.complexity
                total_functions += visitor.function_count

            except Exception:
                continue

        return total_complexity / max(total_functions, 1)

    def _analyze_code_duplication(self) -> float:
        """分析代码重复率"""
        all_lines = []
        duplicate_lines = 0

        for py_file in self.project_path.rglob("*.py"):
            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    lines = f.readlines()

                for line in lines:
                    stripped_line = line.strip()
                    if len(stripped_line) > 20 and not stripped_line.startswith('#'):
                        if stripped_line in all_lines:
                            duplicate_lines += 1
                        else:
                            all_lines.append(stripped_line)

            except Exception:
                continue

        total_lines = len(all_lines)
        return (duplicate_lines / max(total_lines, 1)) * 100

    def _analyze_coupling(self) -> float:
        """分析耦合度"""
        import_graph = {}

        for py_file in self.project_path.rglob("*.py"):
            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                tree = ast.parse(content)

                class ImportVisitor(ast.NodeVisitor):
                    def __init__(self, file_path):
                        self.file_path = file_path
                        self.imports = []

                    def visit_Import(self, node):
                        for alias in node.names:
                            self.imports.append(alias.name)
                        self.generic_visit(node)

                    def visit_ImportFrom(self, node):
                        if node.module:
                            self.imports.append(node.module)
                        self.generic_visit(node)

                visitor = ImportVisitor(py_file)
                visitor.visit(tree)

                import_graph[str(py_file)] = visitor.imports

            except Exception:
                continue

        # 计算耦合度
        total_coupling = 0
        for file_path, imports in import_graph.items():
            total_coupling += len(imports)

        return total_coupling / len(import_graph)

    def _analyze_cohesion(self) -> float:
        """分析内聚性"""
        cohesion_scores = []

        for py_file in self.project_path.rglob("*.py"):
            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                tree = ast.parse(content)

                class CohesionVisitor(ast.NodeVisitor):
                    def __init__(self):
                        self.classes = []
                        self.functions = []

                    def visit_ClassDef(self, node):
                        self.classes.append(node)
                        self.generic_visit(node)

                    def visit_FunctionDef(self, node):
                        self.functions.append(node)
                        self.generic_visit(node)

                visitor = CohesionVisitor()
                visitor.visit(tree)

                # 计算类内聚性
                for class_node in visitor.classes:
                    cohesion_score = self._calculate_class_cohesion(class_node)
                    cohesion_scores.append(cohesion_score)

            except Exception:
                continue

        return sum(cohesion_scores) / max(len(cohesion_scores), 1)

    def _calculate_class_cohesion(self, class_node: ast.ClassDef) -> float:
        """计算类内聚性"""
        methods = [node for node in class_node.body if isinstance(node, ast.FunctionDef)]
        attributes = [node for node in class_node.body if isinstance(node, ast.Assign)]

        if len(methods) == 0:
            return 1.0

        # 简化的内聚性计算
        method_count = len(methods)
        attribute_count = len(attributes)

        # 理想情况下，每个方法都应该使用类的属性
        cohesion = min(1.0, attribute_count / max(method_count, 1))
        return cohesion

    def _analyze_documentation(self) -> float:
        """分析文档覆盖率"""
        total_functions = 0
        documented_functions = 0

        for py_file in self.project_path.rglob("*.py"):
            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                tree = ast.parse(content)

                class DocVisitor(ast.NodeVisitor):
                    def __init__(self):
                        self.functions = 0
                        self.documented = 0

                    def visit_FunctionDef(self, node):
                        self.functions += 1
                        if (node.body and
                            isinstance(node.body[0], ast.Expr) and
                            isinstance(node.body[0].value, ast.Constant) and
                            isinstance(node.body[0].value.value, str)):
                            self.documented += 1
                        self.generic_visit(node)

                visitor = DocVisitor()
                visitor.visit(tree)

                total_functions += visitor.functions
                documented_functions += visitor.documented

            except Exception:
                continue

        return (documented_functions / max(total_functions, 1)) * 100

    def _analyze_test_coverage(self) -> float:
        """分析测试覆盖率"""
        test_files = list(self.project_path.rglob("test_*.py"))
        source_files = list(self.project_path.rglob("*.py"))

        # 简化的测试覆盖率计算
        test_count = len(test_files)
        source_count = len(source_files)

        return (test_count / max(source_count, 1)) * 100

    def _analyze_technical_debt(self) -> float:
        """分析技术债务"""
        debt_indicators = 0
        total_files = 0

        for py_file in self.project_path.rglob("*.py"):
            total_files += 1

            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                # 检查技术债务指标
                if "TODO" in content:
                    debt_indicators += 1
                if "FIXME" in content:
                    debt_indicators += 1
                if "HACK" in content:
                    debt_indicators += 1
                if "XXX" in content:
                    debt_indicators += 1

            except Exception:
                continue

        return (debt_indicators / max(total_files, 1)) * 100

    def generate_report(self) -> str:
        """生成可维护性报告"""
        report = []
        report.append("=" * 80)
        report.append("VirtualChemLab 可维护性评估报告")
        report.append("=" * 80)
        report.append(f"分析时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append("")

        # 指标详情
        report.append("## 可维护性指标")
        report.append("-" * 40)
        report.append(f"圈复杂度: {self.metrics.cyclomatic_complexity:.2f}")
        report.append(f"代码重复率: {self.metrics.code_duplication:.2f}%")
        report.append(f"耦合度: {self.metrics.coupling_coefficient:.2f}")
        report.append(f"内聚性: {self.metrics.cohesion_score:.2f}")
        report.append(f"文档覆盖率: {self.metrics.documentation_coverage:.2f}%")
        report.append(f"测试覆盖率: {self.metrics.test_coverage:.2f}%")
        report.append(f"技术债务: {self.metrics.technical_debt:.2f}%")
        report.append("")

        # 评级
        overall_score = self._calculate_overall_score()
        report.append(f"## 总体评分: {overall_score:.2f}/100")
        report.append(f"评级: {self._get_rating(overall_score)}")

        # 改进建议

        if self.metrics.cyclomatic_complexity > 10:

        if self.metrics.code_duplication > 20:

    def _calculate_overall_score(self) -> float:
        """计算总体评分"""
        # 权重分配
        weights = {
            'cyclomatic_complexity': 0.2,
            'code_duplication': 0.15,
            'coupling_coefficient': 0.15,
            'cohesion_score': 0.15,
            'documentation_coverage': 0.15,
            'test_coverage': 0.15,
            'technical_debt': 0.05
        }

        # 标准化分数
        scores = {
            'cyclomatic_complexity': max(0, 100 - self.metrics.cyclomatic_complexity * 5),
            'code_duplication': max(0, 100 - self.metrics.code_duplication),
            'coupling_coefficient': max(0, 100 - self.metrics.coupling_coefficient * 3),
            'cohesion_score': self.metrics.cohesion_score * 100,
            'documentation_coverage': self.metrics.documentation_coverage,
            'test_coverage': self.metrics.test_coverage,
            'technical_debt': max(0, 100 - self.metrics.technical_debt)
        }

        # 加权平均
        overall_score = sum(scores[key] * weights[key] for key in weights)
        return overall_score
```

```python
# 模块化架构
from typing import Dict, Any
from dataclasses import dataclass
from enum import Enum

class ModuleType(Enum):
    """模块类型"""
    CORE = "core"
    FEATURE = "feature"
    INFRASTRUCTURE = "infrastructure"
    UI = "ui"

@dataclass
class Module:
    """模块定义"""
    name: str
    type: ModuleType
    dependencies: List[str]
    interfaces: List[str]
    implementation: str

class ModularArchitecture:
    """模块化架构管理器"""

    def __init__(self):
        self.modules: Dict[str, Module] = {}
        self.dependency_graph: Dict[str, List[str]] = {}

    def register_module(self, module: Module) -> None:
        """注册模块"""
        self.modules[module.name] = module
        self.dependency_graph[module.name] = module.dependencies

    def get_module(self, name: str) -> Module:
        """获取模块"""
        return self.modules.get(name)

    def resolve_dependencies(self, module_name: str) -> List[str]:
        """解析依赖"""
        visited = set()
        result = []

        def dfs(current):
            if current in visited:
                return

            visited.add(current)

            for dependency in self.dependency_graph.get(current, []):
                dfs(dependency)

            result.append(current)

        dfs(module_name)
        return result

    def validate_architecture(self) -> List[str]:
        """验证架构"""
        errors = []

        # 检查循环依赖
        for module_name in self.modules:
            if self._has_circular_dependency(module_name):
                errors.append(f"模块 {module_name} 存在循环依赖")

        # 检查未定义的依赖
        for module_name, dependencies in self.dependency_graph.items():
            for dependency in dependencies:
                if dependency not in self.modules:
                    errors.append(f"模块 {module_name} 依赖未定义的模块 {dependency}")

        return errors

    def _has_circular_dependency(self, module_name: str) -> bool:
        """检查循环依赖"""
        visited = set()
        rec_stack = set()

        def dfs(current):
            visited.add(current)
            rec_stack.add(current)

            for dependency in self.dependency_graph.get(current, []):
                if dependency not in visited:
                    if dfs(dependency):
                        return True
                elif dependency in rec_stack:
                    return True

            rec_stack.remove(current)
            return False

        return dfs(module_name)
```

```python
# 增强的依赖注入容器
from typing import TypeVar, Type, Callable, Any, Dict, List
from abc import ABC, abstractmethod
import inspect

T = TypeVar('T')

class ServiceLifetime(Enum):
    """服务生命周期"""
    SINGLETON = "singleton"
    TRANSIENT = "transient"
    SCOPED = "scoped"

class ServiceDescriptor:
    """服务描述符"""

    def __init__(self,
                 service_type: Type[T],
                 implementation: Type[T] = None,
                 factory: Callable[[], T] = None,
                 instance: T = None,
                 lifetime: ServiceLifetime = ServiceLifetime.TRANSIENT):
        self.service_type = service_type
        self.implementation = implementation
        self.factory = factory
        self.instance = instance
        self.lifetime = lifetime

class EnhancedDIContainer:
    """增强的依赖注入容器"""

    def __init__(self):
        self._services: Dict[Type, ServiceDescriptor] = {}
        self._scoped_instances: Dict[Type, Any] = {}
        self._current_scope: str = None

    def register_singleton(self, service_type: Type[T], implementation: Type[T] = None) -> 'EnhancedDIContainer':
        """注册单例服务"""
        self._services[service_type] = ServiceDescriptor(
            service_type=service_type,
            implementation=implementation,
            lifetime=ServiceLifetime.SINGLETON
        )
        return self

    def register_transient(self, service_type: Type[T], implementation: Type[T] = None) -> 'EnhancedDIContainer':
        """注册瞬态服务"""
        self._services[service_type] = ServiceDescriptor(
            service_type=service_type,
            implementation=implementation,
            lifetime=ServiceLifetime.TRANSIENT
        )
        return self

    def register_scoped(self, service_type: Type[T], implementation: Type[T] = None) -> 'EnhancedDIContainer':
        """注册作用域服务"""
        self._services[service_type] = ServiceDescriptor(
            service_type=service_type,
            implementation=implementation,
            lifetime=ServiceLifetime.SCOPED
        )
        return self

    def register_factory(self, service_type: Type[T], factory: Callable[[], T]) -> 'EnhancedDIContainer':
        """注册工厂服务"""
        self._services[service_type] = ServiceDescriptor(
            service_type=service_type,
            factory=factory,
            lifetime=ServiceLifetime.TRANSIENT
        )
        return self

    def register_instance(self, service_type: Type[T], instance: T) -> 'EnhancedDIContainer':
        """注册实例服务"""
        self._services[service_type] = ServiceDescriptor(
            service_type=service_type,
            instance=instance,
            lifetime=ServiceLifetime.SINGLETON
        )
        return self

    def resolve(self, service_type: Type[T]) -> T:
        """解析服务"""
        if service_type not in self._services:
            raise ValueError(f"服务未注册: {service_type}")

        descriptor = self._services[service_type]

        # 检查生命周期
        if descriptor.lifetime == ServiceLifetime.SINGLETON:
            if descriptor.instance is None:
                descriptor.instance = self._create_instance(descriptor)
            return descriptor.instance

        elif descriptor.lifetime == ServiceLifetime.SCOPED:
            if self._current_scope is None:
                raise ValueError("没有活动的作用域")

            scope_key = f"{self._current_scope}:{service_type}"
            if scope_key not in self._scoped_instances:
                self._scoped_instances[scope_key] = self._create_instance(descriptor)
            return self._scoped_instances[scope_key]

        else:  # TRANSIENT
            return self._create_instance(descriptor)

    def _create_instance(self, descriptor: ServiceDescriptor) -> Any:
        """创建实例"""
        if descriptor.instance is not None:
            return descriptor.instance

        if descriptor.factory is not None:
            return descriptor.factory()

        if descriptor.implementation is not None:
            return self._create_with_dependencies(descriptor.implementation)

        return self._create_with_dependencies(descriptor.service_type)

    def _create_with_dependencies(self, cls: Type[T]) -> T:
        """创建带依赖的实例"""
        try:
            # 获取构造函数签名
            sig = inspect.signature(cls.__init__)

            # 解析参数
            kwargs = {}
            for param_name, param in sig.parameters.items():
                if param_name == 'self':
                    continue

                # 尝试从容器解析依赖
                if param.annotation != inspect.Parameter.empty:
                    try:
                        kwargs[param_name] = self.resolve(param.annotation)
                    except ValueError:
                        # 如果无法解析，使用默认值
                        if param.default != inspect.Parameter.empty:
                            kwargs[param_name] = param.default
                        else:
                            raise ValueError(f"无法解析依赖: {param_name}")

            return cls(**kwargs)

        except Exception as e:
            raise ValueError(f"创建实例失败 {cls}: {e}")

    def create_scope(self, scope_name: str) -> 'ServiceScope':
        """创建作用域"""
        return ServiceScope(self, scope_name)

    def validate_registrations(self) -> List[str]:
        """验证注册"""
        errors = []

        for service_type, descriptor in self._services.items():
            # 检查循环依赖
            if self._has_circular_dependency(service_type):
                errors.append(f"服务 {service_type} 存在循环依赖")

            # 检查依赖是否可解析
            if descriptor.implementation:
                try:
                    self._validate_dependencies(descriptor.implementation)
                except Exception as e:
                    errors.append(f"服务 {service_type} 依赖验证失败: {e}")

        return errors

    def _has_circular_dependency(self, service_type: Type) -> bool:
        """检查循环依赖"""
        visited = set()
        rec_stack = set()

        def dfs(current):
            visited.add(current)
            rec_stack.add(current)

            if current in self._services:
                descriptor = self._services[current]
                if descriptor.implementation:
                    # 检查构造函数的依赖
                    sig = inspect.signature(descriptor.implementation.__init__)
                    for param_name, param in sig.parameters.items():
                        if param_name == 'self':
                            continue
                        if param.annotation != inspect.Parameter.empty:
                            if param.annotation not in visited:
                                if dfs(param.annotation):
                                    return True
                            elif param.annotation in rec_stack:
                                return True

            rec_stack.remove(current)
            return False

        return dfs(service_type)

    def _validate_dependencies(self, cls: Type) -> None:
        """验证依赖"""
        sig = inspect.signature(cls.__init__)
        for param_name, param in sig.parameters.items():
            if param_name == 'self':
                continue
            if param.annotation != inspect.Parameter.empty:
                if param.annotation not in self._services:
                    raise ValueError(f"未注册的依赖: {param.annotation}")

class ServiceScope:
    """服务作用域"""

    def __init__(self, container: EnhancedDIContainer, scope_name: str):
        self.container = container
        self.scope_name = scope_name
        self._previous_scope = container._current_scope

    def __enter__(self):
        self.container._current_scope = self.scope_name
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        # 清理作用域实例
        scope_keys = [key for key in self.container._scoped_instances.keys()
                     if key.startswith(f"{self.scope_name}:")]
        for key in scope_keys:
            del self.container._scoped_instances[key]

        self.container._current_scope = self._previous_scope

# 使用示例
def setup_container() -> EnhancedDIContainer:
    """设置容器"""
    container = EnhancedDIContainer()

    # 注册服务
    container.register_singleton(ExperimentRepository, ExperimentRepositoryImpl)
    container.register_singleton(TemplateEngine, TemplateEngineImpl)
    container.register_singleton(RuleValidator, RuleValidatorImpl)
    container.register_transient(ExperimentUseCase, ExperimentUseCaseImpl)

    # 验证注册
    errors = container.validate_registrations()
    if errors:
        raise ValueError(f"容器配置错误: {errors}")

    return container
```

```python
        doc.append("")

        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()

            tree = ast.parse(content)

            class APIVisitor(ast.NodeVisitor):
                def __init__(self):
                    self.classes = []
                    self.functions = []

                def visit_ClassDef(self, node):
                    self.classes.append(node)
                    self.generic_visit(node)

                def visit_FunctionDef(self, node):
                    if not node.name.startswith('_'):
                        self.functions.append(node)
                    self.generic_visit(node)

            visitor = APIVisitor()
            visitor.visit(tree)

            # 生成类文档
            for class_node in visitor.classes:
                doc.extend(self._generate_class_doc(class_node))

            # 生成函数文档
            for func_node in visitor.functions:
                doc.extend(self._generate_function_doc(func_node))
```

```python
from pathlib import Path
import ast
import subprocess

class TestCoverageAnalyzer:
    """测试覆盖率分析器"""

    def __init__(self, project_path: Path):
        self.project_path = project_path
        self.src_path = project_path / "src"
        self.tests_path = project_path / "tests"

    def analyze_coverage(self) -> Dict[str, Any]:
        """分析测试覆盖率"""
        # 1. 运行测试覆盖率
        coverage_data = self._run_coverage()

        # 2. 分析未覆盖的代码
        uncovered_code = self._find_uncovered_code()

        # 3. 生成测试建议
        test_suggestions = self._generate_test_suggestions()

        return {
            "coverage_percentage": coverage_data.get("coverage_percentage", 0),
            "uncovered_files": uncovered_code,
            "test_suggestions": test_suggestions
        }

    def _run_coverage(self) -> Dict[str, Any]:
        """运行测试覆盖率"""
        try:
            # 运行pytest with coverage
            result = subprocess.run([
                "python", "-m", "pytest",
                "--cov=src",
                "--cov-report=json",
                "--cov-report=html"
            ], capture_output=True, text=True)

            if result.returncode == 0:
                # 解析覆盖率报告
                coverage_file = self.project_path / "coverage.json"
                if coverage_file.exists():
                    import json
                    with open(coverage_file, 'r') as f:
                        coverage_data = json.load(f)

                    return {
                        "coverage_percentage": coverage_data.get("totals", {}).get("percent_covered", 0),
                        "covered_files": coverage_data.get("files", {})
                    }

            return {"coverage_percentage": 0, "covered_files": {}}

        except Exception as e:
            print(f"运行覆盖率测试失败: {e}")
            return {"coverage_percentage": 0, "covered_files": {}}

    def _find_uncovered_code(self) -> List[Dict[str, Any]]:
        """查找未覆盖的代码"""
        uncovered = []

        for py_file in self.src_path.rglob("*.py"):
            if py_file.name == "__init__.py":
                continue

            try:
                with open(py_file, 'r', encoding='utf-8') as f:
                    content = f.read()

                tree = ast.parse(content)

                class UncoveredVisitor(ast.NodeVisitor):
                    def __init__(self):
                        self.uncovered_functions = []
                        self.uncovered_classes = []

                    def visit_FunctionDef(self, node):
                        # 检查是否有测试
                        if not self._has_test_for_function(node.name, py_file):
                            self.uncovered_functions.append({
                                "name": node.name,
                                "line": node.lineno,
                                "type": "function"
                            })
                        self.generic_visit(node)

                    def visit_ClassDef(self, node):
                        # 检查是否有测试
                        if not self._has_test_for_class(node.name, py_file):
                            self.uncovered_classes.append({
                                "name": node.name,
                                "line": node.lineno,
                                "type": "class"
                            })
                        self.generic_visit(node)

                    def _has_test_for_function(self, func_name: str, file_path: Path) -> bool:
                        """检查函数是否有测试"""
                        test_file = self.tests_path / f"test_{file_path.stem}.py"
                        if test_file.exists():
                            with open(test_file, 'r', encoding='utf-8') as f:
                                test_content = f.read()
                            return func_name in test_content
                        return False

                    def _has_test_for_class(self, class_name: str, file_path: Path) -> bool:
                        """检查类是否有测试"""
                        test_file = self.tests_path / f"test_{file_path.stem}.py"
                        if test_file.exists():
                            with open(test_file, 'r', encoding='utf-8') as f:
                                test_content = f.read()
                            return class_name in test_content
                        return False

                visitor = UncoveredVisitor()
                visitor.visit(tree)

                if visitor.uncovered_functions or visitor.uncovered_classes:
                    uncovered.append({
                        "file": str(py_file),
                        "uncovered_functions": visitor.uncovered_functions,
                        "uncovered_classes": visitor.uncovered_classes
                    })

            except Exception as e:
                print(f"分析文件失败 {py_file}: {e}")

        return uncovered

    def _generate_test_suggestions(self) -> List[str]:
        """生成测试建议"""
        suggestions = []

        # 分析项目结构，生成测试建议
        for py_file in self.src_path.rglob("*.py"):
            if py_file.name == "__init__.py":
                continue

            test_file = self.tests_path / f"test_{py_file.stem}.py"
            if not test_file.exists():
                suggestions.append(f"为 {py_file} 创建测试文件: {test_file}")

        return suggestions

    def generate_test_templates(self) -> None:
        """生成测试模板"""
        for py_file in self.src_path.rglob("*.py"):
            if py_file.name == "__init__.py":
                continue

            test_file = self.tests_path / f"test_{py_file.stem}.py"
            if not test_file.exists():
                self._create_test_template(py_file, test_file)

    def _create_test_template(self, source_file: Path, test_file: Path) -> None:
        """创建测试模板"""
        try:
            with open(source_file, 'r', encoding='utf-8') as f:
                content = f.read()

            tree = ast.parse(content)

            class TestTemplateVisitor(ast.NodeVisitor):
                def __init__(self):
                    self.classes = []
                    self.functions = []

                def visit_ClassDef(self, node):
                    self.classes.append(node)
                    self.generic_visit(node)

                def visit_FunctionDef(self, node):
                    if not node.name.startswith('_'):
                        self.functions.append(node)
                    self.generic_visit(node)

            visitor = TestTemplateVisitor()
            visitor.visit(tree)

            # 生成测试模板
            test_content = []
            test_content.append(f'"""测试 {source_file.name}"""')
            test_content.append("")
            test_content.append("import pytest")
            test_content.append(f"from {source_file.parent.name}.{source_file.stem} import *")
            test_content.append("")

            # 为类生成测试
            for class_node in visitor.classes:
                test_content.append(f"class Test{class_node.name}:")
                test_content.append('    """测试 {class_node.name} 类"""')
                test_content.append("")
                test_content.append("    def test_init(self):")
                test_content.append("        \"\"\"测试初始化\"\"\"")
                test_content.append("        # TODO: 实现测试")
                test_content.append("        pass")
                test_content.append("")

            # 为函数生成测试
            for func_node in visitor.functions:
                test_content.append(f"def test_{func_node.name}():")
                test_content.append(f'    """测试 {func_node.name} 函数"""')
                test_content.append("    # TODO: 实现测试")
                test_content.append("    pass")
                test_content.append("")

            # 保存测试文件
            test_file.parent.mkdir(parents=True, exist_ok=True)
            test_file.write_text("\n".join(test_content))

        except Exception as e:
            print(f"创建测试模板失败 {source_file}: {e}")
```
