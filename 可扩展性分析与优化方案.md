class MicroserviceRegistry:

        self.service_clients: Dict[str, 'ServiceClient'] = {}

    def register_service(self, service_info: ServiceInfo) -> None:
        """注册服务"""
        self.services[service_info.name] = service_info
        self.service_clients[service_info.name] = ServiceClient(service_info)

    def discover_service(self, service_name: str) -> ServiceInfo:
        """发现服务"""
        return self.services.get(service_name)

    def get_service_client(self, service_name: str) -> 'ServiceClient':
        """获取服务客户端"""
        return self.service_clients.get(service_name)

    async def health_check_all(self) -> Dict[str, bool]:
        """健康检查所有服务"""
        results = {}

        for service_name, service_info in self.services.items():
            try:
                client = self.service_clients[service_name]
                results[service_name] = await client.health_check()
            except Exception:
                results[service_name] = False

        return results

class ServiceClient:
    """服务客户端"""

    def __init__(self, service_info: ServiceInfo):
        self.service_info = service_info
        self.base_url = f"http://{service_info.host}:{service_info.port}"
        self.session = None

    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()

    async def call_service(self, endpoint: str, method: str = "GET",
                          data: Dict[str, Any] = None) -> Dict[str, Any]:
        """调用服务"""
        url = f"{self.base_url}/{endpoint}"

        async with self.session.request(method, url, json=data) as response:
            return await response.json()

    async def health_check(self) -> bool:
        """健康检查"""
        try:
            async with self.session.get(f"{self.base_url}/health") as response:
                return response.status == 200
        except Exception:
            return False

from typing import Dict, List, Any, Optional
import asyncio
import aiohttp

class StorageNode:
    """存储节点"""

    def __init__(self, node_id: str, host: str, port: int, capacity: int):
        self.node_id = node_id
        self.host = host
        self.port = port
        self.capacity = capacity
        self.used_space = 0
        self.status = "active"

    def can_store(self, size: int) -> bool:
        """检查是否可以存储"""
        return self.status == "active" and (self.used_space + size) <= self.capacity

    def store(self, key: str, data: bytes) -> bool:
        """存储数据"""
        if not self.can_store(len(data)):
            return False

        # 实际存储逻辑
        self.used_space += len(data)
        return True

    def retrieve(self, key: str) -> Optional[bytes]:
        """检索数据"""
        if self.status != "active":
            return None

        # 实际检索逻辑
        return b"retrieved_data"

    def delete(self, key: str) -> bool:
        """删除数据"""
        if self.status != "active":
            return False

        # 实际删除逻辑
        return True

class DistributedStorage:
    """分布式存储"""

    def __init__(self, replication_factor: int = 3):
        self.nodes: Dict[str, StorageNode] = {}
        self.replication_factor = replication_factor
        self.consistent_hash = ConsistentHash()

    def add_node(self, node: StorageNode) -> None:
        """添加存储节点"""
        self.nodes[node.node_id] = node
        self.consistent_hash.add_node(node.node_id)

    def remove_node(self, node_id: str) -> None:
        """移除存储节点"""
        if node_id in self.nodes:
            del self.nodes[node_id]
            self.consistent_hash.remove_node(node_id)

    def store(self, key: str, data: bytes) -> bool:
        """存储数据"""
        # 计算数据哈希
        data_hash = hashlib.md5(data).hexdigest()

        # 选择存储节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        success_count = 0
        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                if node.store(key, data):
                    success_count += 1

        return success_count >= (self.replication_factor // 2 + 1)

    def retrieve(self, key: str) -> Optional[bytes]:
        """检索数据"""
        # 选择存储节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                data = node.retrieve(key)
                if data is not None:
                    return data

        return None

    def delete(self, key: str) -> bool:
        """删除数据"""
        # 选择存储节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        success_count = 0
        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                if node.delete(key):
                    success_count += 1

        return success_count > 0

class ConsistentHash:
    """一致性哈希"""

    def __init__(self, virtual_nodes: int = 150):
        self.virtual_nodes = virtual_nodes
        self.ring = {}
        self.sorted_keys = []

    def add_node(self, node_id: str) -> None:
        """添加节点"""
        for i in range(self.virtual_nodes):
            virtual_key = f"{node_id}:{i}"
            hash_value = self._hash(virtual_key)
            self.ring[hash_value] = node_id
            self.sorted_keys.append(hash_value)

        self.sorted_keys.sort()

    def remove_node(self, node_id: str) -> None:
        """移除节点"""
        keys_to_remove = []
        for key, value in self.ring.items():
            if value == node_id:
                keys_to_remove.append(key)

        for key in keys_to_remove:
            del self.ring[key]
            self.sorted_keys.remove(key)

    def get_nodes(self, key: str, count: int) -> List[str]:
        """获取节点"""
        if not self.sorted_keys:
            return []

        hash_value = self._hash(key)
        idx = self._find_index(hash_value)

        nodes = []
        for i in range(count):
            ring_key = self.sorted_keys[idx % len(self.sorted_keys)]
            node_id = self.ring[ring_key]
            if node_id not in nodes:
                nodes.append(node_id)
            idx += 1

        return nodes

    def _hash(self, key: str) -> int:
        """计算哈希值"""
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

    def _find_index(self, hash_value: int) -> int:
        """查找索引"""
        for i, key in enumerate(self.sorted_keys):
            if key >= hash_value:
                return i
        return 0

# 分布式缓存实现
from typing import Dict, List, Any, Optional
import asyncio
import aiohttp
import json
import time
from dataclasses import dataclass
from enum import Enum

class CacheNode:
    """缓存节点"""

    def __init__(self, node_id: str, host: str, port: int):
        self.node_id = node_id
        self.host = host
        self.port = port
        self.cache: Dict[str, 'CacheEntry'] = {}
        self.status = "active"
        self.hit_count = 0
        self.miss_count = 0

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        if key in self.cache:
            entry = self.cache[key]
            if not entry.is_expired():
                self.hit_count += 1
                entry.update_access()
                return entry.value
            else:
                del self.cache[key]

        self.miss_count += 1
        return None

    def set(self, key: str, value: Any, ttl: int = 3600) -> None:
        """设置缓存"""
        entry = CacheEntry(value, ttl)
        self.cache[key] = entry

    def delete(self, key: str) -> bool:
        """删除缓存"""
        if key in self.cache:
            del self.cache[key]
            return True
        return False

    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        total_requests = self.hit_count + self.miss_count
        hit_rate = self.hit_count / total_requests if total_requests > 0 else 0

        return {
            "node_id": self.node_id,
            "status": self.status,
            "cache_size": len(self.cache),
            "hit_count": self.hit_count,
            "miss_count": self.miss_count,
            "hit_rate": hit_rate
        }

@dataclass
class CacheEntry:
    """缓存条目"""
    value: Any
    ttl: int
    created_at: float = None
    last_accessed: float = None
    access_count: int = 0

    def __post_init__(self):
        if self.created_at is None:
            self.created_at = time.time()
        if self.last_accessed is None:
            self.last_accessed = time.time()

    def is_expired(self) -> bool:
        """检查是否过期"""
        return time.time() - self.created_at > self.ttl

    def update_access(self) -> None:
        """更新访问信息"""
        self.last_accessed = time.time()
        self.access_count += 1

class DistributedCache:
    """分布式缓存"""

    def __init__(self, replication_factor: int = 2):
        self.nodes: Dict[str, CacheNode] = {}
        self.replication_factor = replication_factor
        self.consistent_hash = ConsistentHash()
        self.default_ttl = 3600

    def add_node(self, node: CacheNode) -> None:
        """添加缓存节点"""
        self.nodes[node.node_id] = node
        self.consistent_hash.add_node(node.node_id)

    def remove_node(self, node_id: str) -> None:
        """移除缓存节点"""
        if node_id in self.nodes:
            del self.nodes[node_id]
            self.consistent_hash.remove_node(node_id)

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        # 选择缓存节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                value = node.get(key)
                if value is not None:
                    return value

        return None

    def set(self, key: str, value: Any, ttl: int = None) -> bool:
        """设置缓存"""
        if ttl is None:
            ttl = self.default_ttl

        # 选择缓存节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        success_count = 0
        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                node.set(key, value, ttl)
                success_count += 1

        return success_count > 0

    def delete(self, key: str) -> bool:
        """删除缓存"""
        # 选择缓存节点
        primary_nodes = self.consistent_hash.get_nodes(key, self.replication_factor)

        success_count = 0
        for node_id in primary_nodes:
            if node_id in self.nodes:
                node = self.nodes[node_id]
                if node.delete(key):
                    success_count += 1

        return success_count > 0

    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        stats = {
            "nodes": {},
            "total_hit_count": 0,
            "total_miss_count": 0,
            "overall_hit_rate": 0.0
        }

        for node_id, node in self.nodes.items():
            node_stats = node.get_stats()
            stats["nodes"][node_id] = node_stats
            stats["total_hit_count"] += node_stats["hit_count"]
            stats["total_miss_count"] += node_stats["miss_count"]

        total_requests = stats["total_hit_count"] + stats["total_miss_count"]
        if total_requests > 0:
            stats["overall_hit_rate"] = stats["total_hit_count"] / total_requests

        return stats

    def cleanup_expired(self) -> int:
        """清理过期缓存"""
        total_cleaned = 0

        for node in self.nodes.values():
            expired_keys = []
            for key, entry in node.cache.items():
                if entry.is_expired():
                    expired_keys.append(key)

            for key in expired_keys:
                del node.cache[key]
                total_cleaned += 1

        return total_cleaned

# 缓存策略优化
from typing import Dict, List, Any, Optional, Callable
from dataclasses import dataclass
from enum import Enum
import time
import threading
from collections import OrderedDict

class CacheStrategy(Enum):
    """缓存策略"""
    LRU = "lru"
    LFU = "lfu"
    FIFO = "fifo"
    TTL = "ttl"
    ADAPTIVE = "adaptive"

@dataclass
class CacheMetrics:
    """缓存指标"""
    hit_count: int = 0
    miss_count: int = 0
    eviction_count: int = 0
    total_size: int = 0
    memory_usage: int = 0

class AdaptiveCache:
    """自适应缓存"""

    def __init__(self, max_size: int = 1000, initial_strategy: CacheStrategy = CacheStrategy.LRU):
        self.max_size = max_size
        self.current_strategy = initial_strategy
        self.cache: OrderedDict = OrderedDict()
        self.access_times: Dict[str, float] = {}
        self.access_counts: Dict[str, int] = {}
        self.metrics = CacheMetrics()
        self.lock = threading.RLock()
        self.strategy_history: List[CacheStrategy] = []
        self.performance_history: List[float] = []

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        with self.lock:
            if key in self.cache:
                entry = self.cache[key]

                # 检查是否过期
                if entry.is_expired():
                    del self.cache[key]
                    self.metrics.miss_count += 1
                    return None

                # 更新访问信息
                self.access_times[key] = time.time()
                self.access_counts[key] = self.access_counts.get(key, 0) + 1

                # 移动到末尾（LRU）
                self.cache.move_to_end(key)

                self.metrics.hit_count += 1
                return entry.value

            self.metrics.miss_count += 1
            return None

    def set(self, key: str, value: Any, ttl: int = 3600) -> None:
        """设置缓存"""
        with self.lock:
            # 检查是否需要驱逐
            if len(self.cache) >= self.max_size and key not in self.cache:
                self._evict_entry()

            # 创建缓存条目
            entry = CacheEntry(value, ttl)
            self.cache[key] = entry
            self.access_times[key] = time.time()
            self.access_counts[key] = 1

            # 更新指标
            self.metrics.total_size = len(self.cache)

    def _evict_entry(self) -> str:
        """驱逐缓存条目"""
        if not self.cache:
            return None

        key_to_evict = None

        if self.current_strategy == CacheStrategy.LRU:
            # 最近最少使用
            key_to_evict = next(iter(self.cache))

        elif self.current_strategy == CacheStrategy.LFU:
            # 最少使用
            min_count = float('inf')
            for key, count in self.access_counts.items():
                if count < min_count:
                    min_count = count
                    key_to_evict = key

        elif self.current_strategy == CacheStrategy.FIFO:
            # 先进先出
            key_to_evict = next(iter(self.cache))

        elif self.current_strategy == CacheStrategy.TTL:
            # 基于TTL
            oldest_time = float('inf')
            for key, entry in self.cache.items():
                if entry.created_at < oldest_time:
                    oldest_time = entry.created_at
                    key_to_evict = key

        elif self.current_strategy == CacheStrategy.ADAPTIVE:
            # 自适应策略
            key_to_evict = self._adaptive_eviction()

        if key_to_evict:
            del self.cache[key_to_evict]
            self.access_times.pop(key_to_evict, None)
            self.access_counts.pop(key_to_evict, None)
            self.metrics.eviction_count += 1

        return key_to_evict

    def _adaptive_eviction(self) -> str:
        """自适应驱逐策略"""
        # 分析访问模式
        current_time = time.time()
        recent_accesses = 0
        total_accesses = 0

        for key, access_time in self.access_times.items():
            if current_time - access_time < 300:  # 5分钟内
                recent_accesses += 1
            total_accesses += 1

        # 如果最近访问较多，使用LRU
        if recent_accesses / max(total_accesses, 1) > 0.7:
            return next(iter(self.cache))

        # 如果访问频率差异较大，使用LFU
        if len(set(self.access_counts.values())) > len(self.access_counts) * 0.5:
            min_count = min(self.access_counts.values())
            for key, count in self.access_counts.items():
                if count == min_count:
                    return key

        # 默认使用LRU
        return next(iter(self.cache))

    def adapt_strategy(self) -> None:
        """自适应调整策略"""
        if len(self.performance_history) < 10:
            return

        # 计算当前策略的性能
        current_performance = self.get_hit_rate()
        self.performance_history.append(current_performance)

        # 如果性能下降，尝试其他策略
        if len(self.performance_history) >= 5:
            recent_avg = sum(self.performance_history[-5:]) / 5
            overall_avg = sum(self.performance_history) / len(self.performance_history)

            if recent_avg < overall_avg * 0.9:  # 性能下降10%
                self._switch_strategy()

    def _switch_strategy(self) -> None:
        """切换策略"""
        current_strategy = self.current_strategy
        self.strategy_history.append(current_strategy)

        # 选择下一个策略
        strategies = [s for s in CacheStrategy if s != current_strategy]
        if strategies:
            self.current_strategy = strategies[0]
            print(f"切换缓存策略: {current_strategy.value} -> {self.current_strategy.value}")

    def get_hit_rate(self) -> float:
        """获取命中率"""
        total_requests = self.metrics.hit_count + self.metrics.miss_count
        return self.metrics.hit_count / max(total_requests, 1)

    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        return {
            "strategy": self.current_strategy.value,
            "hit_rate": self.get_hit_rate(),
            "total_size": self.metrics.total_size,
            "hit_count": self.metrics.hit_count,
            "miss_count": self.metrics.miss_count,
            "eviction_count": self.metrics.eviction_count,
            "strategy_history": [s.value for s in self.strategy_history[-10:]]
        }

    def cleanup_expired(self) -> int:
        """清理过期缓存"""
        with self.lock:
            expired_keys = []
            for key, entry in self.cache.items():
                if entry.is_expired():
                    expired_keys.append(key)

            for key in expired_keys:
                del self.cache[key]
                self.access_times.pop(key, None)
                self.access_counts.pop(key, None)

            return len(expired_keys)

from typing import Dict, List, Any, Optional, Type, Callable
from abc import ABC, abstractmethod

class PluginType(Enum):
    """插件类型"""

    VISUALIZATION = "visualization"
    ANALYSIS = "analysis"
    EXPORT = "export"
    INTEGRATION = "integration"

class PluginStatus(Enum):
    """插件状态"""
    INSTALLED = "installed"
    ENABLED = "enabled"
    DISABLED = "disabled"
    ERROR = "error"
    LOADING = "loading"

@dataclass
class PluginInfo:

    version: str
    description: str
    author: str
    plugin_type: PluginType

class PluginInterface(ABC):

    @abstractmethod

    @abstractmethod
    def get_supported_formats(self) -> List[str]:

        pass

class PluginManager:
    """插件管理器"""

    def __init__(self, plugin_directory: Path):
        self.plugin_directory = plugin_directory
        self.plugins: Dict[str, PluginInterface] = {}
        self.plugin_info: Dict[str, PluginInfo] = {}
        self.dependency_graph: Dict[str, List[str]] = {}
        self.plugin_registry: Dict[PluginType, List[str]] = {}
        self.config_manager = PluginConfigManager()

    def discover_plugins(self) -> List[PluginInfo]:
        """发现插件"""
        discovered_plugins = []

        for plugin_path in self.plugin_directory.iterdir():
            if plugin_path.is_dir() and (plugin_path / "plugin.yaml").exists():
                try:
                    plugin_info = self._load_plugin_info(plugin_path)
                    discovered_plugins.append(plugin_info)
                except Exception as e:
                    print(f"加载插件信息失败 {plugin_path}: {e}")

        return discovered_plugins

    def _load_plugin_info(self, plugin_path: Path) -> PluginInfo:
        """加载插件信息"""
        config_file = plugin_path / "plugin.yaml"

        with open(config_file, 'r', encoding='utf-8') as f:
            config = yaml.safe_load(f)

        return PluginInfo(
            name=config["name"],
            version=config["version"],
            description=config["description"],
            author=config["author"],
            plugin_type=PluginType(config["type"]),
            dependencies=config.get("dependencies", []),
            status=PluginStatus.INSTALLED,
            path=str(plugin_path),
            config_schema=config.get("config_schema"),
            metadata=config.get("metadata", {})
        )

    def install_plugin(self, plugin_info: PluginInfo) -> bool:
        """安装插件"""
        try:
            # 检查依赖
            if not self._check_dependencies(plugin_info):
                return False

            # 加载插件模块
            plugin_module = self._load_plugin_module(plugin_info)
            if not plugin_module:
                return False

            # 创建插件实例
            plugin_instance = self._create_plugin_instance(plugin_info, plugin_module)
            if not plugin_instance:
                return False

            # 验证配置
            config = self.config_manager.get_config(plugin_info.name)
            if not plugin_instance.validate_config(config):
                return False

            # 初始化插件
            if not plugin_instance.initialize(config):
                return False

            # 注册插件
            self.plugins[plugin_info.name] = plugin_instance
            self.plugin_info[plugin_info.name] = plugin_info

            # 更新注册表
            if plugin_info.plugin_type not in self.plugin_registry:
                self.plugin_registry[plugin_info.plugin_type] = []
            self.plugin_registry[plugin_info.plugin_type].append(plugin_info.name)

            # 更新依赖图
            self.dependency_graph[plugin_info.name] = plugin_info.dependencies

            plugin_info.status = PluginStatus.ENABLED
            return True

        except Exception as e:
            print(f"安装插件失败 {plugin_info.name}: {e}")
            plugin_info.status = PluginStatus.ERROR
            return False

    def _check_dependencies(self, plugin_info: PluginInfo) -> bool:
        """检查依赖"""
        for dependency in plugin_info.dependencies:
            if dependency not in self.plugins:
                print(f"插件 {plugin_info.name} 缺少依赖: {dependency}")
                return False

        return True

    def _load_plugin_module(self, plugin_info: PluginInfo) -> Optional[Any]:
        """加载插件模块"""
        try:
            plugin_path = Path(plugin_info.path)
            module_name = f"plugins.{plugin_info.name}"

            # 添加插件路径到sys.path
            import sys
            if str(plugin_path.parent) not in sys.path:
                sys.path.insert(0, str(plugin_path.parent))

            # 导入插件模块
            module = importlib.import_module(module_name)
            return module

        except Exception as e:
            print(f"加载插件模块失败 {plugin_info.name}: {e}")
            return None

    def _create_plugin_instance(self, plugin_info: PluginInfo, module: Any) -> Optional[PluginInterface]:
        """创建插件实例"""
        try:
            # 查找插件类
            plugin_class = None
            for name, obj in inspect.getmembers(module):
                if (inspect.isclass(obj) and
                    issubclass(obj, PluginInterface) and
                    obj != PluginInterface):
                    plugin_class = obj
                    break

            if not plugin_class:
                print(f"未找到插件类 {plugin_info.name}")
                return None

            # 创建实例
            instance = plugin_class()
            return instance

        except Exception as e:
            print(f"创建插件实例失败 {plugin_info.name}: {e}")
            return None

    def uninstall_plugin(self, plugin_name: str) -> bool:
        """卸载插件"""
        if plugin_name not in self.plugins:
            return False

        try:
            # 清理插件
            plugin = self.plugins[plugin_name]
            plugin.cleanup()

            # 移除插件
            del self.plugins[plugin_name]
            plugin_info = self.plugin_info[plugin_name]
            plugin_info.status = PluginStatus.DISABLED

            # 更新注册表
            plugin_type = plugin_info.plugin_type
            if plugin_type in self.plugin_registry:
                self.plugin_registry[plugin_type].remove(plugin_name)

            # 更新依赖图
            if plugin_name in self.dependency_graph:
                del self.dependency_graph[plugin_name]

            return True

        except Exception as e:
            print(f"卸载插件失败 {plugin_name}: {e}")
            return False

    def get_plugin(self, plugin_name: str) -> Optional[PluginInterface]:
        """获取插件"""
        return self.plugins.get(plugin_name)

    def get_plugins_by_type(self, plugin_type: PluginType) -> List[PluginInterface]:
        """按类型获取插件"""
        plugin_names = self.plugin_registry.get(plugin_type, [])
        return [self.plugins[name] for name in plugin_names if name in self.plugins]

    def validate_dependencies(self) -> List[str]:
        """验证依赖"""
        errors = []

        for plugin_name, dependencies in self.dependency_graph.items():
            for dependency in dependencies:
                if dependency not in self.plugins:
                    errors.append(f"插件 {plugin_name} 缺少依赖: {dependency}")

        return errors

    def get_plugin_stats(self) -> Dict[str, Any]:
        """获取插件统计"""
        stats = {
            "total_plugins": len(self.plugins),
            "enabled_plugins": len([p for p in self.plugin_info.values() if p.status == PluginStatus.ENABLED]),
            "disabled_plugins": len([p for p in self.plugin_info.values() if p.status == PluginStatus.DISABLED]),
            "error_plugins": len([p for p in self.plugin_info.values() if p.status == PluginStatus.ERROR]),
            "plugins_by_type": {}
        }

        for plugin_type in PluginType:
            count = len(self.plugin_registry.get(plugin_type, []))
            stats["plugins_by_type"][plugin_type.value] = count

        return stats

# 异步处理系统实现
from typing import Dict, List, Any, Optional, Callable
import asyncio
import aiohttp
from dataclasses import dataclass
from enum import Enum
import time
import threading
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

class TaskStatus(Enum):
    """任务状态"""
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"

@dataclass
class Task:
    """任务"""
    id: str
    name: str
    func: Callable
    args: tuple
    kwargs: dict
    status: TaskStatus
    result: Any = None
    error: Optional[Exception] = None
    created_at: float = None
    started_at: Optional[float] = None
    completed_at: Optional[float] = None

    def __post_init__(self):
        if self.created_at is None:
            self.created_at = time.time()

class AsyncTaskManager:
    """异步任务管理器"""

    def __init__(self, max_workers: int = 10):
        self.max_workers = max_workers
        self.tasks: Dict[str, Task] = {}
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.workers: List[asyncio.Task] = []
        self.thread_pool = ThreadPoolExecutor(max_workers=max_workers)
        self.process_pool = ProcessPoolExecutor(max_workers=max_workers)
        self.running = False
        self.lock = asyncio.Lock()

    async def start(self) -> None:
        """启动任务管理器"""
        if self.running:
            return

        self.running = True

        # 启动工作线程
        for i in range(self.max_workers):
            worker = asyncio.create_task(self._worker(f"worker-{i}"))
            self.workers.append(worker)

        print(f"异步任务管理器已启动，工作线程数: {self.max_workers}")

    async def stop(self) -> None:
        """停止任务管理器"""
        if not self.running:
            return

        self.running = False

        # 取消所有工作线程
        for worker in self.workers:
            worker.cancel()

        # 等待所有工作线程完成
        await asyncio.gather(*self.workers, return_exceptions=True)

        # 关闭线程池和进程池
        self.thread_pool.shutdown(wait=True)
        self.process_pool.shutdown(wait=True)

        print("异步任务管理器已停止")

    async def submit_task(self, name: str, func: Callable, *args, **kwargs) -> str:
        """提交任务"""
        task_id = str(uuid.uuid4())
        task = Task(
            id=task_id,
            name=name,
            func=func,
            args=args,
            kwargs=kwargs,
            status=TaskStatus.PENDING
        )

        async with self.lock:
            self.tasks[task_id] = task
            await self.task_queue.put(task)

        return task_id

    async def get_task_result(self, task_id: str) -> Any:
        """获取任务结果"""
        async with self.lock:
            task = self.tasks.get(task_id)
            if not task:
                raise ValueError(f"任务不存在: {task_id}")

            if task.status == TaskStatus.COMPLETED:
                return task.result
            elif task.status == TaskStatus.FAILED:
                raise task.error
            else:
                return None

    async def wait_for_task(self, task_id: str, timeout: float = None) -> Any:
        """等待任务完成"""
        async with self.lock:
            task = self.tasks.get(task_id)
            if not task:
                raise ValueError(f"任务不存在: {task_id}")

        # 等待任务完成
        start_time = time.time()
        while task.status in [TaskStatus.PENDING, TaskStatus.RUNNING]:
            if timeout and (time.time() - start_time) > timeout:
                raise TimeoutError(f"任务超时: {task_id}")

            await asyncio.sleep(0.1)

        if task.status == TaskStatus.COMPLETED:
            return task.result
        elif task.status == TaskStatus.FAILED:
            raise task.error
        else:
            raise RuntimeError(f"任务状态异常: {task.status}")

    async def cancel_task(self, task_id: str) -> bool:
        """取消任务"""
        async with self.lock:
            task = self.tasks.get(task_id)
            if not task:
                return False

            if task.status in [TaskStatus.PENDING, TaskStatus.RUNNING]:
                task.status = TaskStatus.CANCELLED
                return True

            return False

    async def _worker(self, worker_name: str) -> None:
        """工作线程"""
        print(f"工作线程 {worker_name} 已启动")

        while self.running:
            try:
                # 获取任务
                task = await asyncio.wait_for(self.task_queue.get(), timeout=1.0)

                # 执行任务
                await self._execute_task(task)

            except asyncio.TimeoutError:
                continue
            except Exception as e:
                print(f"工作线程 {worker_name} 错误: {e}")

        print(f"工作线程 {worker_name} 已停止")

    async def _execute_task(self, task: Task) -> None:
        """执行任务"""
        try:
            # 更新任务状态
            task.status = TaskStatus.RUNNING
            task.started_at = time.time()

            # 执行任务
            if asyncio.iscoroutinefunction(task.func):
                # 异步函数
                result = await task.func(*task.args, **task.kwargs)
            else:
                # 同步函数，在线程池中执行
                loop = asyncio.get_event_loop()
                result = await loop.run_in_executor(
                    self.thread_pool,
                    lambda: task.func(*task.args, **task.kwargs)
                )

            # 更新任务结果
            task.result = result
            task.status = TaskStatus.COMPLETED
            task.completed_at = time.time()

        except Exception as e:
            # 更新任务错误
            task.error = e
            task.status = TaskStatus.FAILED
            task.completed_at = time.time()

    def get_task_stats(self) -> Dict[str, Any]:
        """获取任务统计"""
        stats = {
            "total_tasks": len(self.tasks),
            "pending_tasks": len([t for t in self.tasks.values() if t.status == TaskStatus.PENDING]),
            "running_tasks": len([t for t in self.tasks.values() if t.status == TaskStatus.RUNNING]),
            "completed_tasks": len([t for t in self.tasks.values() if t.status == TaskStatus.COMPLETED]),
            "failed_tasks": len([t for t in self.tasks.values() if t.status == TaskStatus.FAILED]),
            "cancelled_tasks": len([t for t in self.tasks.values() if t.status == TaskStatus.CANCELLED]),
            "queue_size": self.task_queue.qsize()
        }

        return stats

    def cleanup_completed_tasks(self, max_age: float = 3600) -> int:
        """清理已完成的任务"""
        current_time = time.time()
        tasks_to_remove = []

        for task_id, task in self.tasks.items():
            if (task.status in [TaskStatus.COMPLETED, TaskStatus.FAILED, TaskStatus.CANCELLED] and
                task.completed_at and (current_time - task.completed_at) > max_age):
                tasks_to_remove.append(task_id)

        for task_id in tasks_to_remove:
            del self.tasks[task_id]

        return len(tasks_to_remove)

# 使用示例
async def setup_async_system():
    """设置异步系统"""
    task_manager = AsyncTaskManager(max_workers=5)

    # 启动任务管理器
    await task_manager.start()

    # 提交任务
    def cpu_intensive_task(n: int) -> int:
        """CPU密集型任务"""
        result = 0
        for i in range(n):
            result += i
        return result

    task_id = await task_manager.submit_task("cpu_task", cpu_intensive_task, 1000000)

    # 等待任务完成
    result = await task_manager.wait_for_task(task_id)
    print(f"任务结果: {result}")

    # 获取统计信息
    stats = task_manager.get_task_stats()
    print(f"任务统计: {stats}")

    # 停止任务管理器
    await task_manager.stop()

    return task_manager
